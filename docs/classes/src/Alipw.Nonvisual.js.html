1 /** 
  2 * @constructor 
  3 * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  4 * @description 所有非可视类的基类。
  5 */
  6 Alipw.Nonvisual = function(){
  7 	
  8 	var pub = function(config){
  9 		this.config = config || (new Object());
 10 		this.commitProperties(config);
 11 		if(this.autoInit){
 12 			this.initialize();
 13 		}
 14 	};
 15 	
 16 	pub.prototype = 
 17 	/**
 18 	 * @lends Alipw.Nonvisual.prototype
 19 	 */ 
 20 	{
 21 		/**
 22 		 * @property
 23 		 * @type Boolean
 24 		 * @description [config]自动初始化。在实例化对象时自动初始化。
 25 		 * @default true
 26 		 */
 27 		autoInit:true,
 28 		/**
 29 		 * @property
 30 		 * @type Boolean
 31 		 * @description 指示该对象是否已被初始化。
 32 		 */
 33 		initialized:false,
 34 		/**
 35 		 * @property
 36 		 * @type Boolean
 37 		 * @description 指示该对象当前是否可用。
 38 		 */
 39 		enabled:true,
 40 		/**
 41 		 * @property
 42 		 * @type Object
 43 		 * @default null
 44 		 * @description [config option]定义事件监听。
 45 		 */
 46 		listeners:null,
 47 		commitProperties:function(){
 48 			Alipw.apply(this, this.config);
 49 		},
 50 		/**
 51 		 * @public
 52 		 * @description 对Nonvisual实例进行初始化
 53 		 */
 54 		initialize:function(){
 55 			this.evtProxy = jQuery(new Object());
 56 			this.initialized = true;
 57 			
 58 			if(this.listeners){
 59 				for(var i in this.listeners){
 60 					if(this.listeners[i] instanceof Function){
 61 						this.addEventListener(i,this.listeners[i],this);
 62 					}
 63 				}
 64 			}
 65 		},
 66 		/**
 67 		 * @public
 68 		 * @description 添加事件监听
 69 		 * @param {String} type 事件类型。
 70 		 * @param {Function} fn 监听函数。
 71 		 * @param {Object} [scope] 作用域。指定监听函数fn中的this指针所指向的对象。默认为undefined。
 72 		 */
 73 		addEventListener:function(type, fn, scope){
 74 			Alipw.EventManager.addListener(this,type,fn,scope,false);
 75 		},
 76 		/**
 77 		 * @public
 78 		 * @description 移除事件监听
 79 		 * @param {String} type 事件类型。
 80 		 * @param {Function} fn 监听函数。
 81 		 */
 82 		removeEventListener:function(type, fn){
 83 			Alipw.EventManager.removeListener(this,type,fn,false);
 84 		},
 85 		/**
 86 		 * @public
 87 		 * @description 触发指定事件
 88 		 * @param {String} type 事件类型。
 89 		 * @param {Object} [params] 附带参数。
 90 		 */
 91 		fireEvent:function(type,params){
 92 			return Alipw.EventManager.fireEvent(this,type,params,false,false);
 93 		},
 94 		regEvents:function(events){
 95 			
 96 		},
 97 		update:function(){
 98 			
 99 		},
100 		/**
101 		 * @public
102 		 * @description 销毁对象，释放创建的资源与引用
103 		 */
104 		destroy:function(){
105 			this.fireEvent("destroy");
106 		},
107 		/**
108 		 * @public
109 		 * @description 启用
110 		 */
111 		enable:function(){
112 			this.enabled = true;
113 		},
114 		/**
115 		 * @public
116 		 * @description 禁用
117 		 */
118 		disable:function(){
119 			this.enabled = false;
120 		}
121 	};
122 	
123 	return pub;
124 }();
