1 /**
  2  * @constructor Alipw.Button
  3  * @extends Alipw.BoxComponent
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 按钮。支持九宫格方式的按钮皮肤。
  6  * @example
  7  * 
  8  */
  9 
 10 Alipw.Button = Alipw.extend(Alipw.BoxComponent,
 11 /** @lends Alipw.Button.prototype */
 12 {
 13 	/**
 14 	 * @property
 15 	 * @default null
 16 	 * @type Number
 17 	 * @description [config option]定义点击事件的监听函数。
 18 	 */
 19 	handler:null,
 20 	/**
 21 	 * @property
 22 	 * @default null
 23 	 * @type Object
 24 	 * @description [config option]定义点击事件的监听函数中的this指针所指向的对象。
 25 	 */
 26 	scope:null,
 27 	/**
 28 	 * @property
 29 	 * @default 22
 30 	 * @type Number
 31 	 * @description [config option]定义最小高度。
 32 	 */
 33 	minHeight:22,
 34 	/**
 35 	 * @property
 36 	 * @default 10
 37 	 * @type Number
 38 	 * @description [config option]定义最小宽度。
 39 	 */
 40 	minWidth:10,
 41 	/**
 42 	 * @property
 43 	 * @default 65
 44 	 * @type Number
 45 	 * @description [config option]定义最大高度。
 46 	 */
 47 	maxHeight:65,
 48 	/**
 49 	 * @property
 50 	 * @default ''
 51 	 * @type String
 52 	 * @description [config option]定义按钮图标的class名称。
 53 	 */
 54 	iconCls:"",
 55 	/**
 56 	 * @property
 57 	 * @default 'Button'
 58 	 * @type String
 59 	 * @description [config option]定义按钮的文本。
 60 	 */
 61 	label:"Button",
 62 	/**
 63 	 * @property
 64 	 * @default 'alipw-button'
 65 	 * @type String
 66 	 * @description [config option]HTML元素的class名称的基本前缀。
 67 	 */
 68 	baseCls:"alipw-button",
 69 	/**
 70 	 * @property
 71 	 * @type String
 72 	 * @default 'a'
 73 	 * @description [config option]默认的HTML元素tag name。
 74 	 */
 75 	defaultEl:"a",
 76 	constructor:function(){
 77 		Alipw.Button.superclass.constructor.apply(this,arguments);
 78 	},
 79 	commitProperties:function(){
 80 		Alipw.Button.superclass.commitProperties.apply(this,arguments);
 81 	},
 82 	initialize:function(){
 83 		Alipw.Button.superclass.initialize.apply(this,arguments);
 84 	},
 85 	createDom:function(){
 86 		Alipw.Button.superclass.createDom.apply(this,arguments);
 87 		this.el.attr("href","#");
 88 		this.el.append(['<table>',
 89 		    '<tr class="' + this.baseCls + '-top">',
 90 		    	'<td class="' + this.baseCls + '-left"></td>',
 91 		        '<td class="' + this.baseCls + '-center"></td>',
 92 		        '<td class="' + this.baseCls + '-right"></td>',
 93 		    '</tr>',
 94 		    '<tr class="' + this.baseCls + '-middle">',
 95 		    	'<td class="' + this.baseCls + '-left"></td>',
 96 		        '<td class="' + this.baseCls + '-center">',
 97 		        	'<span class="' + this.baseCls + '-icon"></span>',
 98 		        	'<span class="' + this.baseCls + '-text"></span>',
 99 		        '</td>',
100 		        '<td class="' + this.baseCls + '-right"></td>',
101 		    '</tr>',
102 		    '<tr class="' + this.baseCls + '-bottom">',
103 		    	'<td class="' + this.baseCls + '-left"></td>',
104 		        '<td class="' + this.baseCls + '-center"></td>',
105 		        '<td class="' + this.baseCls + '-right"></td>',
106 		    '</tr>',
107 		'</table>'].join(""));
108 		
109 		this.addEventListener("click",this.defaultClickHandler,this,true);
110 		
111 		if(this.handler){
112 			this.addEventListener("click",this.handler,this.scope || this,true);
113 		}
114 	},
115 	renderComplete:function(){
116 		Alipw.Button.superclass.renderComplete.apply(this,arguments);
117 		
118 		this.el.find("." + this.baseCls + "-text").html(this.label);
119 		
120 		this.setIconCls(this.iconCls);
121 		
122 		if(this.label){
123 			this.el.find("." + this.baseCls + "-text").show();
124 		}else{
125 			this.el.find("." + this.baseCls + "-text").hide();
126 		}
127 	},
128 	//protected
129 	_doLayout:function(){
130 		Alipw.Button.superclass._doLayout.apply(this,arguments);
131 		if(this.width){
132 			this.el.find("." + this.baseCls + "-middle " + "." + this.baseCls + "-center").width(
133 					this.width -
134 					this.el.find("." + this.baseCls + "-left").width() -
135 					this.el.find("." + this.baseCls + "-right").width()
136 			);
137 		}
138 		
139 		if(this.height){
140 			this.el.find("." + this.baseCls + "-middle " + "." + this.baseCls + "-center").height(
141 					this.height -
142 					this.el.find("." + this.baseCls + "-top").height() -
143 					this.el.find("." + this.baseCls + "-bottom").height()
144 			);
145 		}
146 	},
147 	/**
148 	 * @public
149 	 * @description 设定Button的图标的class名称。
150 	 * @param {String} cls class名称。
151 	 */
152 	setIconCls:function(cls){
153 		var iconEl = this.el.find("." + this.baseCls + "-icon");
154 		if(this.iconCls){
155 			iconEl.removeClass(this.iconCls);
156 		}
157 		this.iconCls = cls;
158 		iconEl.addClass(cls);
159 
160 		if(cls){
161 			iconEl.show();
162 		}else{
163 			iconEl.hide();
164 		}
165 	},
166 	defaultClickHandler:function(e){
167 		e.preventDefault();
168 	}
169 });
