1 /**
  2  * @constructor Alipw.Validator
  3  * @extends Alipw.BoxComponent
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 表单验证工具。
  6  * @example
  7  * var validator = new Alipw.Validator({
  8  *	target:"textbox",
  9  *	msgPosition:"right",
 10  *	conditions:[{
 11  *		exp:"abc",
 12  *		errorMsg:"requires 'abc'",
 13  *	},{
 14  *		exp:"rainy",
 15  *		errorMsg:"requires 'rainy'",
 16  *	}],
 17  *	correctMsg:"correct string!",
 18  *	triggerEvents:["keyup","blur"]
 19  *});
 20  * 
 21  */
 22 
 23 Alipw.Validator = Alipw.extend(Alipw.BoxComponent,
 24 /** @lends Alipw.Validator.prototype */
 25 {
 26 	zIndex:100,
 27 	/**
 28 	 * @property
 29 	 * @type Array
 30 	 * */
 31 	conditions:null,
 32 	/**
 33 	 * @property
 34 	 * */
 35 	correctMsg:"Correct!",
 36 	/**
 37 	 * @property
 38 	 * @description the position of the message box. defaults to "right". this configure is invalid only when the renderTo is not set.
 39 	 * */
 40 	msgPosition:"right",
 41 	/**
 42 	 * @property
 43 	 * @description the target which is validated
 44 	 * */
 45 	target:null,
 46 	/**
 47 	 * @property
 48 	 * @description 
 49 	 * */
 50 	targetAttr:"value",
 51 	/**
 52 	 * @property
 53 	 * @description the event that fires the validation
 54 	 * */
 55 	triggerEvents:"change",
 56 	baseCls:"alipw-validator",
 57 	/**
 58 	 * @property
 59 	 * @description defaults to false
 60 	 * */
 61 	autoRender:false,
 62 	/**
 63 	 * @property
 64 	 * @description [read only] tells if the specific value passed the validation
 65 	 * */
 66 	success:false,
 67 	isPending:false,
 68 	pendingIndex:null,
 69 	constructor:function(){
 70 		Alipw.Validator.superclass.constructor.apply(this,arguments);
 71 	},
 72 	initialize:function(){
 73 		Alipw.Validator.superclass.initialize.apply(this,arguments);
 74 	},
 75 	commitProperties:function(){
 76 		Alipw.Validator.superclass.commitProperties.apply(this,arguments);
 77 		
 78 		this.targetEl = Alipw.convertEl(this.target);
 79 		
 80 		if(typeof(this.triggerEvents)=="string"){
 81 			this.triggerEvents = [this.triggerEvents];
 82 		}
 83 		
 84 		for(var i=0,len=this.triggerEvents.length;i<len;i++){
 85 			this.targetEl.bind(this.triggerEvents[i],jQuery.proxy(this.validate,this));
 86 		}
 87 	},
 88 	createDom:function(){
 89 		Alipw.Validator.superclass.createDom.apply(this,arguments);
 90 		var msgBoxHTML = ['<div class="' + this.baseCls + '-wrap">',
 91 		                  	'<div class="' + this.baseCls + '-errorMsg"></div>',
 92 		                  	'<div class="' + this.baseCls + '-correctMsg">' + this.correctMsg + '</div>',
 93 		                  	'<div class="' + this.baseCls + '-pendingMsg"></div>',
 94 		                  '</div>'].join("");
 95 		this.el.append(msgBoxHTML);
 96 	},
 97 	renderComplete:function(){
 98 		Alipw.Validator.superclass.renderComplete.apply(this,arguments);
 99 		
100 		if(!this.renderTo){
101 			this.el.css("position","absolute");
102 			$(window).bind("resize",jQuery.proxy(this.setPosition_Validator,this));
103 		}
104 	},
105 	/**
106 	 * @public
107 	 * @description validate the target
108 	 * @return Boolean isValid
109 	 * */
110 	validate:function(){
111 		if(this.isPending){
112 			return this.success;
113 		}
114 		
115 		if(!this.rendered){
116 			this.render();
117 		}
118 		this.setVisible(true);
119 		if(!this.renderTo){
120 			this.setPosition_Validator();
121 		}
122 		
123 		var wrap = this.el.find("." + this.baseCls + "-wrap");
124 		
125 		
126 		var isCorrect = true;
127 		var i = 0, len = this.conditions.length;
128 		if(Alipw.isSet(this.pendingIndex)){
129 			i = this.pendingIndex + 1;
130 		}
131 		
132 		var faildFn = function(i){
133 			this.isPending = false;
134 			wrap.removeClass(this.baseCls + "-wrap-correct");
135 			wrap.removeClass(this.baseCls + "-wrap-pending");
136 			wrap.addClass(this.baseCls + "-wrap-error");
137 			
138 			this.el.find("." + this.baseCls + "-errorMsg").html(this.conditions[i].errorMsg);
139 			isCorrect = false;
140 			this.pendingIndex = null;
141 			this.success = false;
142 			
143 			this.fireEvent('validateComplete',{},false);
144 		};
145 		for(i;i<len;i++){
146 			//for expression validation
147 			if(this.conditions[i].exp){
148 				if(!this.targetEl.attr(this.targetAttr).match(this.conditions[i].exp)){
149 					wrap.removeClass(this.baseCls + "-wrap-correct");
150 					wrap.removeClass(this.baseCls + "-wrap-pending");
151 					wrap.addClass(this.baseCls + "-wrap-error");
152 					
153 					this.el.find("." + this.baseCls + "-errorMsg").html(this.conditions[i].errorMsg);
154 					isCorrect = false;
155 					break;
156 				}
157 			//for remote validation
158 			}else if(this.conditions[i].url && this.conditions[i].fn){
159 				this.isPending = true;
160 				this.pendingIndex = i;
161 				this.success = 'validating';
162 				wrap.removeClass(this.baseCls + "-wrap-correct");
163 				wrap.removeClass(this.baseCls + "-wrap-error");
164 				wrap.addClass(this.baseCls + "-wrap-pending");
165 				this.el.find("." + this.baseCls + "-pendingMsg").html(this.conditions[i].pendingMsg || '');
166 				
167 				var valueParam = this.conditions[i].valueParamName || 'value';
168 				var ajaxData = new Object();
169 				ajaxData[valueParam] = this.targetEl.attr(this.targetAttr);
170 				jQuery.ajax({
171 					url:this.conditions[i].url,
172 					method:this.conditions[i].method || 'GET',
173 					data:ajaxData,
174 					success:jQuery.proxy(function(response){
175 						this.isPending = false;
176 						if(this.conditions[i].fn.call(this.conditions[i],response) == false){
177 							faildFn.call(this,i);
178 						}else{
179 							this.validate();
180 						}
181 					},this),
182 					error:jQuery.proxy(function(){
183 						faildFn.call(this,i);
184 					},this)
185 				});
186 				return this.success;
187 			}
188 		}
189 		
190 		this.pendingIndex = null;
191 		if(isCorrect){
192 			wrap.removeClass(this.baseCls + "-wrap-pending");
193 			wrap.removeClass(this.baseCls + "-wrap-error");
194 			wrap.addClass(this.baseCls + "-wrap-correct");
195 			this.el.find("." + this.baseCls + "-correctMsg").html(this.correctMsg);
196 			this.success = true;
197 		}else{
198 			this.success = false;
199 		};
200 		
201 		this.fireEvent('validateComplete',{},false);
202 		return this.success;
203 	},
204 	/**
205 	 * @public
206 	 * @description reset the validator
207 	 * */
208 	reset:function(){
209 		this.success = false;
210 		var wrap = this.el.find("." + this.baseCls + "-wrap");
211 		wrap.removeClass(this.baseCls + "-wrap-error");
212 		wrap.removeClass(this.baseCls + "-wrap-correct");
213 		this.setVisible(false);
214 	},
215 	//private
216 	setPosition_Validator:function(){
217 		var x,y;
218 		var targPos = this.targetEl.offset();		
219 		
220 		if(this.msgPosition == "top"){
221 			x = targPos.left;
222 			y = targPos.top - this.el.outerHeight();
223 		}else if (this.msgPosition == "right"){
224 			x = targPos.left + this.targetEl.outerWidth();
225 			y = targPos.top;
226 		}else if(this.msgPosition == "bottom"){
227 			x = targPos.left;
228 			y = targPos.top + this.targetEl.outerHeight();
229 		}else if(this.msgPosition == "left"){
230 			x = targPos.left - this.el.outerWidth();
231 			y = targPos.top;
232 		}
233 		
234 		this.el.css({
235 			left:x+ "px",
236 			top:y + "px"
237 		});
238 	}
239 });
