1 /** 
  2 * @constructor
  3 * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  4 * @description 事件对象。
  5 */
  6 Alipw.Event = function(){
  7 	
  8 	var pub = function(type,object,params,jQueryEvent){
  9 		jQuery.extend(this,params);
 10 		/**
 11 		 * @memberOf Alipw.Event.prototype
 12 		 * @property
 13 		 * @type String
 14 		 * @description 事件类型。
 15 		 */
 16 		this.type = type.replace(/alipw\-/g,'');
 17 		/**
 18 		 * @memberOf Alipw.Event.prototype
 19 		 * @property
 20 		 * @type Object
 21 		 * @description 触发事件的当前对象。
 22 		 */
 23 		this.currentTarget = object;
 24 		/**
 25 		 * @memberOf Alipw.Event.prototype
 26 		 * @property
 27 		 * @type Object
 28 		 * @description 触发事件的原始对象。
 29 		 */
 30 		this.target = object;
 31 		this.jQueryEvent = jQueryEvent;
 32 	};
 33 	
 34 	pub.prototype = 
 35 	/** @lends Alipw.Event.prototype */
 36 	{
 37 		/**
 38 		 * @public
 39 		 * @description 阻止事件响应时浏览器将要发生的默认动作。
 40 		 */
 41 		preventDefault:function(){
 42 			if(this.jQueryEvent){
 43 				this.jQueryEvent.preventDefault();
 44 			}
 45 			this.isDefaultPrevented = true;
 46 		},
 47 		/**
 48 		 * @public
 49 		 * @description 阻止事件的冒泡传播。
 50 		 */
 51 		stopPropagation:function(){
 52 			if(this.jQueryEvent){
 53 				this.jQueryEvent.stopPropagation();
 54 			}
 55 		}
 56 	};
 57 	
 58 	return pub;
 59 }();
