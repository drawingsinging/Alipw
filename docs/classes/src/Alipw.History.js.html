1 /**
  2  * @class
  3  * @type singleton
  4  * @extends Alipw.Nonvisual
  5  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  6  * @description 用于支持浏览器的前进、后退按钮。在浏览器前进和后退按钮被点击、或是手动更改地址栏的hash时，会触发History的change事件。
  7  */
  8 Alipw.History = function(){
  9 	var stack = new Array();
 10 	var iframe;
 11 	var pub = new Alipw.Nonvisual({
 12 		/**
 13 		 * @memberOf Alipw.History
 14 		 * @property
 15 		 * @type Boolean
 16 		 * @description [config]自动初始化。在实例化对象时自动初始化。
 17 		 * @default false
 18 		 */
 19 		autoInit:false
 20 	});
 21 	
 22 	Alipw.apply(pub,
 23 	/** @lends Alipw.History */
 24 	{
 25 		/**
 26 		 * @property
 27 		 * @type String
 28 		 * @description [config option]定义用于支持浏览器前进和后退按钮的空白页面，此URL需要与当前站点在同一域下，用于支持IE6，7下的浏览器按钮动作。如果alipw库的访问地址与站点地址在同一域下，则不需要设置此项。
 29 		 * @default ''
 30 		 */
 31 		blankURL:'',
 32 		currentIndex:0,
 33 		/**
 34 		 * @public
 35 		 * @description 添加一个历史标记到hash中。
 36 		 * @param {String} token 标记名称
 37 		 */
 38 		add:function(token){
 39 			stack.push(token);
 40 			this.currentIndex = stack.length - 1;
 41 			
 42 			//if add action cause hash change,set a flag to ignore the hash change event
 43 			if(window.location.hash.split("#")[1] != token){
 44 				this.status = "adding";
 45 			}
 46 			
 47 			window.location.hash = token;
 48 		},
 49 		forward:function(){
 50 			if(this.iframe){
 51 				this.iframe.contentWindow.history.forward();
 52 			}else{
 53 				history.forward();
 54 			}
 55 		},
 56 		back:function(){
 57 			if(this.iframe){
 58 				this.iframe.contentWindow.history.back();
 59 			}else{
 60 				history.back();
 61 			}
 62 		},
 63 		/**
 64 		 * @public
 65 		 * @description 初始化。在使用History前， 必须先执行此方法，此方法中会自动执行Nonvisual类的initialize方法。
 66 		 */
 67 		init:function(){
 68 			this.initialize();
 69 			
 70 			var me = this;
 71 			Alipw.EventManager.enableHashChangeEvent();
 72 			
 73 			if(Alipw.isIE && parseInt(Alipw.ieVersion) < 8){
 74 				iframe = document.createElement("iframe");
 75 				iframe.settingSrc = true;
 76 				iframe.style.position = "absolute";
 77 				iframe.style.width = "1px";
 78 				iframe.style.height = "1px";
 79 				iframe.style.visibility = "hidden";
 80 				iframe.style.left = "-10000px";
 81 				iframe.style.top = "-10000px";
 82 				var url = this.blankURL?this.blankURL:(Alipw.getPath() + 'alipw/resources/assets/blank.html');
 83 				iframe.src = url + '?' + window.location.hash.replace("#","");
 84 				if(Alipw.isReady){
 85 					document.body.appendChild(iframe);
 86 				}else{
 87 					Alipw.onReady(function(){
 88 						document.body.appendChild(iframe);
 89 					},this);
 90 				}
 91 				
 92 				iframe.onreadystatechange = function(){
 93 					if(iframe.readyState != "interactive"){
 94 						return;
 95 					}
 96 					if(iframe.settingSrc){
 97 						iframe.settingSrc = false;
 98 						return;
 99 					}
100 					
101 					var hash = unescape(iframe.contentWindow.location.href.toString().split("?")[1]);
102 					iframe.changingHash = true;
103 					window.location.hash = hash;
104 				};
105 				
106 				this.iframe = iframe;
107 			}
108 			
109 			Alipw.getWinProxy().addEventListener("hashChange",function(e){
110 				if(parseInt(Alipw.ieVersion) < 8){
111 					if(iframe && iframe.changingHash){
112 						iframe.changingHash = false;
113 					}else{
114 						
115 						//damn ie 6 bug. when changing hash manually in the address bar. 
116 						//then we can't set hash normally by javascript.
117 						if(Alipw.isIE6 && me.status != "adding"){
118 							window.location.reload();
119 						}
120 						
121 						iframe.settingSrc = true;
122 						var url = me.blankURL?me.blankURL:(Alipw.getPath() + 'alipw/resources/assets/blank.html');
123 						iframe.src = url + '?' + window.location.hash.replace("#","");
124 					}
125 				}
126 				
127 				if(me.status=="adding"){
128 					me.status = "added";
129 					return;
130 				}
131 				var evt = new Object();
132 				evt.token = e.hash;
133 				me.fireEvent("change",evt);
134 			});
135 		},
136 		/**
137 		 * @public
138 		 * @description 获取当前浏览器的hash。
139 		 */
140 		getToken:function(){
141 			return window.location.hash.replace("#","");
142 		}
143 	});
144 	
145 	return pub;
146 
147 }();

151 /**
152  * @name Alipw.History#change
153  * @event
154  * @description 在浏览器前进和后退按钮被点击、或是手动更改地址栏的hash而导致hash改变时，会触发History的change事件。
155  * @param {Alipw.Event} e e拥有以下附加属性:<ul><li>token:String 历史标记。</li></ul>
156  */
