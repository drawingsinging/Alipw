1 /** 
  2 * @class
  3 * @type singleton
  4 * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5 * @description 事件管理器。
  6 */
  7 Alipw.EventManager = function(){
  8 	var listenerProxyStore =  new Array();
  9 	
 10 	listenerProxyStore.getFnProxy = function(fn,object){
 11 		for(var i=0,len=listenerProxyStore.length;i<len;i++){
 12 			if(listenerProxyStore[i].fn == fn && listenerProxyStore[i].object == object){
 13 				return listenerProxyStore[i].proxy;
 14 			}
 15 		}
 16 		
 17 		return null;
 18 	};
 19 	
 20 	var pub = 
 21 	/** @lends Alipw.EventManager */
 22 	{
 23 		/**
 24 		 * @public
 25 		 * @description 添加事件监听
 26 		 * @param {Alipw.Component/Alipw.Nonvisual} object 需要监听的对象。
 27 		 * @param {String} type 事件类型。
 28 		 * @param {Function} fn 监听函数。
 29 		 * @param {Object} [scope] 作用域。指定监听函数fn中的this指针所指向的对象。默认为undefined。
 30 		 * @param {Boolean} [useBrowserEvent] 是否为浏览器事件。浏览器事件是指click, mouseover等浏览器自带事件。默认为false。
 31 		 */
 32 		addListener:function(object, type, fn, scope, useBrowserEvent){
 33 			var el;
 34 			if(object instanceof Alipw.Component){
 35 				el = object.el;
 36 			}else if(object instanceof Alipw.Nonvisual){
 37 				el = object.evtProxy;
 38 			}else{
 39 				return;
 40 			}
 41 			
 42 			if(!useBrowserEvent){
 43 				type = "alipw-" + type;
 44 			}
 45 			
 46 			if(useBrowserEvent && object instanceof Alipw.Component){
 47 				el.bind(type,scope?jQuery.proxy(fn,scope):fn);
 48 			}else{
 49 				var fnProxy = listenerProxyStore.getFnProxy(fn,object);
 50 				if(!fnProxy){
 51 					fnProxy = function(jQueryEvent,event){
 52 						event.currentTarget = object;
 53 						event.jQueryEvent = jQueryEvent;
 54 						//to cover the deprecated parameter 'eo' that was used in the past projects,
 55 						//send double event object here.
 56 						fn.call(this,event,event);
 57 					};
 58 					
 59 					listenerProxyStore.push({
 60 						proxy:fnProxy,
 61 						fn:fn,
 62 						object:object
 63 					});
 64 				}
 65 				
 66 				el.bind(type,scope?jQuery.proxy(fnProxy,scope):fnProxy);
 67 			}
 68 		},
 69 		/**
 70 		 * @public
 71 		 * @description 移除事件监听
 72 		 * @param {Alipw.Component/Alipw.Nonvisual} object 需要移除监听的对象。
 73 		 * @param {String} type 事件类型。
 74 		 * @param {Function} fn 监听函数。
 75 		 * @param {Boolean} [useBrowserEvent] 是否为浏览器事件。浏览器事件是指click, mouseover等浏览器自带事件。默认为false。
 76 		 */
 77 		removeListener:function(object, type, fn, useBrowserEvent){
 78 			var el;
 79 			if(object instanceof Alipw.Component){
 80 				el = object.el;
 81 			}else if(object instanceof Alipw.Nonvisual){
 82 				el = object.evtProxy;
 83 			}else{
 84 				return;
 85 			}
 86 			
 87 			if(!useBrowserEvent){
 88 				type = "alipw-" + type;
 89 			}
 90 			
 91 			if(useBrowserEvent && object instanceof Alipw.Component){
 92 				el.unbind(type,fn);
 93 			}else{
 94 				var fnProxy = listenerProxyStore.getFnProxy(fn,object);
 95 				
 96 				if(fnProxy){
 97 					el.unbind(type,fnProxy);
 98 				}
 99 			}
100 		},
101 		/**
102 		 * @public
103 		 * @description 触发该component的指定事件
104 		 * @param {Alipw.Component/Alipw.Nonvisual} object 需要触发事件的对象。
105 		 * @param {String} type 事件类型。
106 		 * @param {Object} [params] 附带参数。
107 		 * @param {Boolean} [bubble] 是否冒泡,默认为true。
108 		 * @param {Boolean} [useBrowserEvent] 是否为浏览器事件。浏览器事件是指click, mouseover等浏览器自带事件。默认为false。
109 		 * @return {Boolean} 如果成功调度了事件，则值为 true。 值 false 表示失败或对事件调用了 preventDefault(). 仅当useBrowserEvent为false时有效。
110 		 */
111 		fireEvent:function(object, type, params, bubble, useBrowserEvent){
112 			var el;
113 			if(object instanceof Alipw.Component){
114 				el = object.el;
115 			}else if(object instanceof Alipw.Nonvisual){
116 				el = object.evtProxy;
117 			}else{
118 				return;
119 			}
120 			
121 			if(!useBrowserEvent){
122 				type = "alipw-" + type;
123 			}
124 			
125 			if(!Alipw.isSet(bubble)){
126 				bubble = true;
127 			}
128 			
129 			if(useBrowserEvent && object instanceof Alipw.Component){
130 				if(bubble){
131 					el.trigger(type,params);
132 				}else{
133 					el.triggerHandler(type,params);
134 				}
135 			}else{
136 				var event = new Alipw.Event(type,object,params);
137 				if(bubble){
138 					el.trigger(type,event);
139 				}else{
140 					el.triggerHandler(type,event);
141 				}
142 				return event.isDefaultPrevented?false:true;
143 			}
144 		},
145 		/**
146 		 * @public
147 		 * @description 开启对hashChange事件的监听
148 		 */
149 		enableHashChangeEvent:function(){
150 			var me = this;
151 			this.hashChangeManager = new Object();
152 			this.hashChangeManager.lastHash = window.location.hash;
153 			this.hashChangeManager.timer = setInterval(function(){
154 				if(window.location.hash !== me.hashChangeManager.lastHash){
155 					var params = new Object();
156 					params.lashHash = me.hashChangeManager.lastHash.replace("#",""); //only remove the first '#'
157 					params.hash = window.location.hash.replace("#","");
158 					Alipw.getWinProxy().fireEvent("hashChange",params);
159 					me.hashChangeManager.lastHash = window.location.hash;
160 				}
161 			},100);
162 		}
163 	};
164 	
165 	return pub;
166 }();
