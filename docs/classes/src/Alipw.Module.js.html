1 /**
  2  * @constructor Alipw.Module
  3  * @extends Alipw.Component
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 模块。模块是一个独立个体。它可以独立的调用一个远程页面，并在指定的位置进行渲染。同时，它还可以调用指定的JavaScript文件并在该模块的资源完全加载完毕后执行。模块还可对在其内部创建的可视控件进行资源管理，在自身被销毁时，通过该模块创建的可视控件也将被销毁。
  6  */
  7 
  8 Alipw.Module = Alipw.extend(Alipw.Component,
  9 /** @lends Alipw.Module.prototype */
 10 {
 11 	/**
 12 	 * @property
 13 	 * @type String
 14 	 * @description [config option]定义模块的HTML模板。此配置仅当TemplateUrl未设置时有效。
 15 	 * @default ''
 16 	 */
 17 	template:'',
 18 	/**
 19 	 * @property
 20 	 * @type String
 21 	 * @description [config option]定义模块的HTML模板的URL。当模块执行load方法时，会调取该URL的返回字符串作为该模块的HTML模板。
 22 	 * @default ''
 23 	 */
 24 	templateUrl:'',
 25 	/**
 26 	 * @property
 27 	 * @type Array
 28 	 * @description [config option]定义模块模板加载完成之后执行的脚本。该数组的元素类型为Function，当模板加载完成后，会依次执行数组中的function，在这些function的执行过程中，this指针指向的对象对当前的module对象。
 29 	 * @Default []
 30 	 */
 31 	scripts:null,
 32 	/**
 33 	 * @property
 34 	 * @type Array
 35 	 * @description [config option]定义模块模板加载完成之后执行的脚本文件url，可实现跨域调用。该数组的元素类型为String，当模板加载完成后，会依次加载数组中的URL作为脚本执行。在这些URL指向的文件中，需要使用Alipw.addModuleScript()方法写入脚本，以保证在调用的脚本中，this指向的对象为该module.
 36 	 * @Default []
 37 	 */
 38 	scriptUrls:null,
 39 	/**
 40 	 * @property
 41 	 * @type String
 42 	 * @description [config option]定义模块处于正在加载的状态时，追加的class名称。
 43 	 * @Default []
 44 	 */
 45 	loadingCls:'',
 46 	commitProperties:function(){
 47 		Alipw.Module.superclass.commitProperties.apply(this,arguments);
 48 	},
 49 	initialize:function(){
 50 		Alipw.Module.superclass.initialize.apply(this,arguments);
 51 		
 52 		if(!this.scripts){
 53 			this.scripts = new Array();
 54 		}
 55 		
 56 		if(!this.scriptUrls){
 57 			this.scriptUrls = new Array();
 58 		}
 59 		
 60 		//module context, read only
 61 		this.context = new Object();
 62 		
 63 		//internal.components in this module.
 64 		this.__componentMgr = new Array();
 65 	},
 66 	createDom:function(){
 67 		Alipw.Module.superclass.createDom.apply(this,arguments);
 68 	},
 69 	renderComplete:function(){
 70 		Alipw.Module.superclass.renderComplete.apply(this,arguments);
 71 	},
 72 	destroy:function(){
 73 		for(var i=0,len=this.__componentMgr.length;i<len;i++){
 74 			this.__componentMgr[i].destroy();
 75 		}
 76 		Alipw.Module.superclass.destroy.apply(this,arguments);
 77 	},
 78 	/**
 79 	 * @public
 80 	 * @description 向模块中增加一段脚本。
 81 	 * @param {Function} fn 增加的脚本片段需要在function包中。
 82 	 */
 83 	addScript:function(fn){
 84 		this.scripts.push(fn);
 85 	},
 86 	/**
 87 	 * @public
 88 	 * @description 加载。如果存在templateUrl,则开始调用远程HTML模板，否则，则使用template作为模板。之后开始调用scriptUrls中URL指向的javascript脚本文件，并在加载完成后依次执行。
 89 	 * @param {Object} arg arg包含以下可设置属性：<ul><li>url : String  可选。模块HTML模板的URL。若为空，则调用模块的templateUrl。</li><li>method : String  调用方式。可选值有'POST','GET'。</li><li>params : Object  提交的参数。可选。</li></ul>
 90 	 */
 91 	load:function(arg){
 92 		this.fireEvent('beforeModuleLoad',{},false);
 93 		
 94 		if(this.loadingCls)this.el.addClass(this.loadingCls);
 95 		this.el.empty();
 96 		for(var i=0,len=this.scripts.length;i<len;i++){
 97 			if(this.scripts[i].__Alipw_needToReload){
 98 				this.scripts.splice(i,1);
 99 				len=this.scripts.length;
100 				i--;
101 			}
102 		}
103 		
104 		if(this.templateUrl){
105 			this.loadTemplate_Module(arg);
106 		}else{
107 			this.loadScripts_Module();
108 		}
109 	},
110 	/**
111 	 * @public
112 	 * @description 在该模块中创建Alipw.Component实例。使用此方法创建实例时，模块会对创建的实例进行管理，当该模块被销毁时，通过该方法所创建的实例也将被销毁。
113 	 * @param {Object(Constructor)} Alipw.Component或其子类。
114 	 * @param {Object} config 初始化实例时的配置对象。
115 	 */
116 	create:function(component,config){
117 		var comp;
118 		if(component){
119 			comp = new component(config);
120 			this.__componentMgr.push(comp);
121 		}
122 		
123 		return comp;
124 	},	
125 	//private
126 	loadTemplate_Module:function(arg){
127 		var _this = this;
128 		if(!arg)arg = new Object();
129 		
130 		var url = arg.url || this.templateUrl;
131 		var method = arg.method || 'GET';
132 		var params = arg.params;
133 		
134 		jQuery.ajax({
135 			url:url,
136 			type:method,
137 			data:params,
138 			success:function(response){
139 				_this.template = response;
140 				_this.loadScripts_Module();
141 			}
142 		});
143 	},
144 	//private
145 	loadScripts_Module:function(){
146 		if(this.destroyed)return;
147 		
148 		var _this = this;
149 		
150 		var len = this.scriptUrls.length;
151 		var loadedNum = 0;
152 		
153 		var loadScript = function(){
154 			if(_this.destroyed)return;
155 			var scriptNode = Alipw.loadScript({
156 				url:_this.scriptUrls[loadedNum],
157 				success:function(){
158 					if(Alipw.Module.__moduleScriptFn instanceof Function){
159 						_this.addScript(Alipw.Module.__moduleScriptFn);
160 						Alipw.Module.__moduleScriptFn = null;
161 					}
162 					
163 					loadedNum++;
164 					if(loadedNum >= len){
165 						_this.renderContent_Module();
166 					}else{
167 						loadScript();
168 					}
169 				}
170 			});
171 			_this.addEventListener('destroy',function(){
172 				jQuery(scriptNode).remove();
173 			});
174 		};
175 		if(len > 0){
176 			loadScript();
177 		}else{
178 			this.renderContent_Module();
179 		}
180 	},
181 	//private
182 	renderContent_Module:function(){
183 		if(this.destroyed)return;
184 		if(this.loadingCls)this.el.removeClass(this.loadingCls);
185 		this.fireEvent('moduleLoaded', {}, false);
186 		
187 		if(this.template){
188 			this.el.append(jQuery(this.template));
189 		}
190 		for(var i=0,len=this.scripts.length;i<len;i++){
191 			if(this.destroyed)return;
192 			this.scripts[i].call(this);
193 		}
194 		
195 		this.fireEvent('moduleRendered', {}, false);
196 	}
197 });
