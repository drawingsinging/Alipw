1 /**
  2  * @constructor Alipw.SlideView
  3  * @extends Alipw.BoxComponent
  4  * @description 滑动播放视图。
  5  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  6  * @example
  7  * 
  8  */
  9 
 10 Alipw.SlideView = Alipw.extend(Alipw.BoxComponent,
 11 /** @lends Alipw.SlideView.prototype */
 12 {
 13 	baseCls:'alipw-slideview',
 14 	width:400,
 15 	height:300,
 16 	subviews:null,
 17 	interval:5000,
 18 	showIndex:true,
 19 	currentIndex:0,
 20 	autoPlay:false,
 21 	transferEasing:'easeInOutExpo',
 22 	transferDuration:500,
 23 	constructor:function(){
 24 		Alipw.SlideView.superclass.constructor.apply(this,arguments);
 25 	},
 26 	commitProperties:function(){
 27 		Alipw.SlideView.superclass.commitProperties.apply(this,arguments);
 28 		if(!this.subviews){
 29 			this.subviews = new Array();
 30 		}
 31 	},
 32 	initialize:function(){
 33 		Alipw.SlideView.superclass.initialize.apply(this,arguments);
 34 	},
 35 	createDom:function(){
 36 		Alipw.SlideView.superclass.createDom.apply(this,arguments);
 37 		this.el.append(new Alipw.Template(
 38 			[
 39 			 '<div class="{$baseCls}-scrollwrap"></div>',
 40 			 '<div class="{$baseCls}-index-bg"></div>',
 41 			 '<div class="{$baseCls}-index"></div>'
 42 			]
 43 		).set({'baseCls':this.baseCls}).compile());
 44 	},
 45 	renderComplete:function(){
 46 		Alipw.SlideView.superclass.renderComplete.apply(this,arguments);
 47 		this.renderSubviews_SlideView();
 48 		this.renderIndex_SlideView();
 49 		if(this.autoPlay){
 50 			this.play();
 51 		}
 52 		
 53 		var scrollwrap = this.el.find('.' + this.baseCls + '-scrollwrap');
 54 		scrollwrap.bind('mouseover',jQuery.proxy(function(){
 55 			clearInterval(this.autoPlayTimer);
 56 		},this));
 57 		scrollwrap.bind('mouseout',jQuery.proxy(function(){
 58 			if(this.isPlaying){
 59 				this.play();
 60 			}
 61 		},this));
 62 	},
 63 	switchTo:function(index){
 64 		clearInterval(this.autoPlayTimer);
 65 		var scrollwrap = this.el.find('.' + this.baseCls + '-scrollwrap');
 66 		var x = this.width * index * -1;
 67 		this.indexBtnItems[this.currentIndex].removeClass(this.baseCls + '-index-selected');
 68 		this.currentIndex = index;
 69 		this.indexBtnItems[index].addClass(this.baseCls + '-index-selected');
 70 		scrollwrap.stop();
 71 		scrollwrap.animate({
 72 			left:x
 73 		},this.transferDuration,this.transferEasing);
 74 		if(this.isPlaying){
 75 			this.play();
 76 		}
 77 	},
 78 	back:function(){
 79 		var index;
 80 		if(this.currentIndex == 0){
 81 			index = this.subviews.length - 1;
 82 		}else{
 83 			index =  this.currentIndex - 1;
 84 		}
 85 		this.switchTo(index);
 86 	},
 87 	forward:function(){
 88 		var index;
 89 		if(this.currentIndex == this.subviews.length - 1){
 90 			index = 0;
 91 		}else{
 92 			index =  this.currentIndex + 1;
 93 		}
 94 		this.switchTo(index);
 95 	},
 96 	//protected
 97 	_doLayout:function(){
 98 		Alipw.SlideView.superclass._doLayout.apply(this,arguments);
 99 	},
100 	//private
101 	renderSubviews_SlideView:function(){
102 		var scrollwrap = this.el.find('.' + this.baseCls + '-scrollwrap');
103 		Alipw.each(this.subviews,function(i,el){
104 			var el = $(el);
105 			var subviewContainer = $('<div class="' + this.baseCls + '-subview"></div>');
106 			subviewContainer.width(this.width);
107 			subviewContainer.height(this.height);
108 			subviewContainer.append(el);
109 			scrollwrap.append(subviewContainer);
110 		},this);
111 	},
112 	//private
113 	renderIndex_SlideView:function(){
114 		var indexContainer = this.el.find('.' + this.baseCls + '-index');
115 		var indexBg = this.el.find('.' + this.baseCls + '-index-bg');
116 		if(!this.showIndex){
117 			indexContainer.hide();
118 			indexBg.hide();
119 		}
120 		this.indexBtnItems = new Array();
121 		Alipw.each(this.subviews,function(i,el){
122 			var indexBtn = $('<a hidefocus="true" href="#">' + (i + 1) + '</a>');
123 			if(i == this.currentIndex){
124 				indexBtn.addClass(this.baseCls + '-index-selected');
125 			}
126 			this.indexBtnItems.push(indexBtn);
127 			indexBtn.click(jQuery.proxy(function(e){
128 				e.preventDefault();
129 				this.switchTo(i);
130 			},this));
131 			if(this.showIndex){
132 				indexContainer.append(indexBtn);
133 			}
134 		},this);
135 	},
136 	play:function(){
137 		var _this = this;
138 		
139 		clearInterval(this.autoPlayTimer);
140 		this.autoPlayTimer = setInterval(function(){
141 			_this.forward();
142 		},this.interval);
143 		this.isPlaying = true;
144 	},
145 	stop:function(){
146 		clearInterval(this.autoPlayTimer);
147 		this.isPlaying = false;
148 	}
149 });
