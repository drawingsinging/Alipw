1 /**
  2  * @constructor
  3  * @extends Alipw.Component
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 盒状可视控件。指具有可自定义宽度和高度的可视控件。
  6  */
  7 
  8 Alipw.BoxComponent = Alipw.extend(Alipw.Component,
  9 /**
 10  * @lends Alipw.BoxComponent.prototype
 11  */
 12 {
 13 	/**
 14 	 * @property
 15 	 * @type Number
 16 	 * @default null
 17 	 * @description [config option]定义BoxComponent的宽度。
 18 	 */
 19 	width:null,
 20 	/**
 21 	 * @property
 22 	 * @type Number
 23 	 * @default null
 24 	 * @description [config option]定义BoxComponent的高度。
 25 	 */
 26 	height:null,
 27 	/**
 28 	 * @property
 29 	 * @type Number
 30 	 * @default 1
 31 	 * @description [config option]定义BoxComponent的最小宽度。
 32 	 */
 33 	minWidth:1,
 34 	/**
 35 	 * @property
 36 	 * @type Number
 37 	 * @default 1
 38 	 * @description [config option]定义BoxComponent的最小高度。
 39 	 */
 40 	minHeight:1,
 41 	/**
 42 	 * @property
 43 	 * @type Number
 44 	 * @default null
 45 	 * @description [config option]定义BoxComponent的最大宽度。
 46 	 */
 47 	maxWidth:null,
 48 	/**
 49 	 * @property
 50 	 * @type Number
 51 	 * @default null
 52 	 * @description [config option]定义BoxComponent的最大高度。
 53 	 */
 54 	maxHeight:null,
 55 	/**
 56 	 * @property
 57 	 * @type Boolean
 58 	 * @description 指示当前是否为自动高度。
 59 	 * @default null
 60 	 */
 61 	autoHeight:null,
 62 	/**
 63 	 * @property
 64 	 * @type Boolean
 65 	 * @default false
 66 	 * @description [config option]定义是否可通过鼠标拖拽来缩放尺寸。
 67 	 */
 68 	resizable:false,
 69 	/**
 70 	 * @property
 71 	 * @type Boolean
 72 	 * @default false
 73 	 * @description [config option]定义是否显示阴影。
 74 	 */
 75 	showShadow:false,
 76 	/**
 77 	 * @property
 78 	 * @type String
 79 	 * @default 'drop'
 80 	 * @description [config option]定义阴影的显示模式，可选值有'drop','frame','sides'。
 81 	 */
 82 	shadowMode:"drop",
 83 	constructor:function(config){
 84 		Alipw.BoxComponent.superclass.constructor.apply(this,arguments);
 85 	},
 86 	commitProperties:function(){
 87 		Alipw.BoxComponent.superclass.commitProperties.apply(this,arguments);
 88 		if(!this.height && !Alipw.isSet(this.autoHeight)){
 89 			this.autoHeight = true;
 90 		}
 91 	},
 92 	initialize:function(){
 93 		Alipw.BoxComponent.superclass.initialize.apply(this,arguments);
 94 	},
 95 	createDom:function(){
 96 		Alipw.BoxComponent.superclass.createDom.apply(this,arguments);
 97 	},
 98 	render:function(obj){
 99 		Alipw.BoxComponent.superclass.render.apply(this,arguments);
100 	},
101 	renderComplete:function(updating){
102 		Alipw.BoxComponent.superclass.renderComplete.apply(this,arguments);
103 		if(this.showShadow){
104 			if(!this.shadow){
105 				this.createShadow_BoxComponent();
106 			}
107 			this.shadow.show();
108 		}else{
109 			if(this.shadow){
110 				this.shadow.setVisible(false);
111 			}
112 		}
113 		
114 		if(this.resizable){
115 			var customizedResizeConfig = typeof(this.resizable) == "object"?this.resizable:{};
116 			this.resizeSupporter = new Alipw.ResizeSupporter(jQuery.extend({
117 				target:this,
118 				maxWidth:this.maxWidth,
119 				maxHeight:this.maxHeight,
120 				minWidth:this.minWidth,
121 				minHeight:this.minHeight
122 			},customizedResizeConfig));
123 			
124 			this.addEventListener('destroy',function(){
125 				if(this.resizeSupporter){
126 					this.resizeSupporter.destroy();
127 					this.resizeSupporter = null;
128 				}
129 			},this);
130 		}
131 	},
132 	//protected
133 	_doLayout:function(){
134 		Alipw.BoxComponent.superclass._doLayout.apply(this,arguments);
135 		if(this.width){
136 			this.el.width(this.width);
137 		}
138 
139 		if(this.autoHeight){
140 			this.height = null;
141 			this.el.css('height','auto');
142 			
143 			var curHeight = this.getHeight();
144 			
145 			if(this.maxHeight && curHeight > this.maxHeight){
146 				this.height = this.maxHeight;
147 				this.el.height(this.maxHeight);
148 			}
149 			
150 			if(this.minHeight && curHeight < this.maxHeight){
151 				this.height = this.minHeight;
152 				this.el.height(this.minHeight);
153 			}
154 			
155 		}else if(this.height){
156 			this.el.height(this.height);
157 		}
158 	},
159 	/**
160 	 * @property
161 	 * @description 设定BoxComponent的宽度。
162 	 * @param {Number} width 以像素为单位的宽度。
163 	 */
164 	setWidth:function(width){
165 		this.width = width;
166 		this.doLayout();
167 		
168 		this.fireEvent("resize",{},false);
169 	},
170 	/**
171 	 * @public
172 	 * @description 设定BoxComponent的高度。
173 	 * @param {Number} width 以像素为单位的高度。
174 	 */
175 	setHeight:function(height){
176 		this.height = height;
177 		this.autoHeight = false;
178 		this.doLayout();
179 		
180 		this.fireEvent("resize",{},false);
181 	},
182 	/**
183 	 * @public
184 	 * @description 设定BoxComponent的宽度和高度。
185 	 * @param {Number} width 以像素为单位的宽度。
186 	 * @param {Number} height 以像素为单位的高度。
187 	 */
188 	setSize:function(width,height){
189 		this.width = width;
190 		this.height = height;
191 		this.doLayout();
192 		
193 		this.fireEvent("resize",{},false);
194 	},
195 	/**
196 	 * @public
197 	 * @description 获取BoxComponent的宽度。
198 	 * @return {Number} 以像素为单位的宽度。
199 	 */
200 	getWidth:function(){
201 		return this.el.innerWidth();
202 	},
203 	/**
204 	 * @public
205 	 * @description 获取BoxComponent的高度。
206 	 * @return {Number} 以像素为单位的高度。
207 	 */
208 	getHeight:function(){
209 		return this.el.innerHeight();
210 	},
211 	//private
212 	createShadow_BoxComponent:function(e){
213 		if(this.shadow){
214 			return;
215 		}
216 		
217 		if(this instanceof Alipw.Shadow){
218 			return;
219 		}
220 		
221 		this.shadow = new Alipw.Shadow({
222 			target:this,
223 			position:this.position,
224 			mode:this.shadowMode
225 		});
226 		this.addEventListener("destroy",function(){
227 			if(this.shadow){
228 				this.shadow.destroy();
229 			}
230 		},this);
231 	}
232 });
