1 /**
  2  * @constructor Alipw.ListItem
  3  * @extends Alipw.BoxComponent
  4  * @description List控件中的列表项目。
  5  * @example
  6  * 
  7  */
  8 
  9 Alipw.ListItem = Alipw.extend(Alipw.BoxComponent,
 10 /** @lends Alipw.ListItem.prototype */
 11 {
 12 	baseCls:'alipw-listitem',
 13 	overCls:'alipw-listitem-over',
 14 	downCls:'alipw-listitem-down',
 15 	selectedCls:'alipw-listitem-selected',
 16 	index:null,
 17 	selected:false,
 18 	autoHeight:false,
 19 	visibleRender:false,
 20 	height:14,
 21 	text:'',
 22 	data:null,
 23 	constructor:function(){
 24 		Alipw.ListItem.superclass.constructor.apply(this,arguments);
 25 	},
 26 	commitProperties:function(){
 27 		Alipw.ListItem.superclass.commitProperties.apply(this,arguments);
 28 	},
 29 	initialize:function(){
 30 		Alipw.ListItem.superclass.initialize.apply(this,arguments);
 31 	},
 32 	createDom:function(){
 33 		Alipw.ListItem.superclass.createDom.apply(this,arguments);
 34 		this.el.append(new Alipw.Template(
 35 			[
 36 			 '<span class="{$baseCls}-text"></span>'
 37 			]
 38 		).set({'baseCls':this.baseCls}).compile());
 39 	},
 40 	renderComplete:function(){
 41 		Alipw.ListItem.superclass.renderComplete.apply(this,arguments);
 42 		this.setText(this.text);
 43 		
 44 		this.addEventListener('click',this.clickHandler,this,true);
 45 	},
 46 	//protected
 47 	_doLayout:function(){
 48 		Alipw.ListItem.superclass._doLayout.apply(this,arguments);
 49 	},
 50 	setText:function(text){
 51 		this.el.find('.' + this.baseCls + '-text').text(this.text);
 52 	},
 53 	select:function(isUserOperation){
 54 		this.el.addClass(this.selectedCls);
 55 		this.selected = true;
 56 		this.fireEvent('select',{
 57 			isUserOperation:isUserOperation
 58 		},false);
 59 	},
 60 	unselect:function(){
 61 		this.el.removeClass(this.selectedCls);
 62 		this.selected = false;
 63 		this.fireEvent('unselect',{},false);
 64 	},
 65 	//private
 66 	clickHandler:function(e){
 67 		e.preventDefault();
 68 		this.fireEvent('itemclick',{},false);
 69 	}
 70 });
