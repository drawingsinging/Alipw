1 /**
  2  * @constructor Alipw.ToolTip
  3  * @extends Alipw.BorderContainer
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 浮动的消息提示。
  6  * @example
  7  * 
  8  */
  9 
 10 Alipw.ToolTip = Alipw.extend(Alipw.BorderContainer,
 11 /** @lends Alipw.ToolTip.prototype */
 12 {
 13 	html:"",
 14 	target:null,
 15 	showDelay:0.5,
 16 	hideDelay:0.2,
 17 	position:"auto",
 18 	floating:true,
 19 	showShadow:true,
 20 	textMaxWidth:300,
 21 	offsetX:0,
 22 	offsetY:20,
 23 	/**
 24 	 * @property
 25 	 * @description the event that fires the validation
 26 	 * */
 27 	triggerEvents:["mouseover"],
 28 	hideEvents:["mouseout"],
 29 	baseCls:"alipw-tooltip",
 30 	/**
 31 	 * @property
 32 	 * @description defaults to false
 33 	 * */
 34 	autoRender:false,
 35 	constructor:function(){
 36 		Alipw.ToolTip.superclass.constructor.apply(this,arguments);
 37 	},
 38 	commitProperties:function(){
 39 		Alipw.ToolTip.superclass.commitProperties.apply(this,arguments);
 40 		
 41 		this.targetEl = Alipw.convertEl(this.target);
 42 	},
 43 	initialize:function(){
 44 		Alipw.ToolTip.superclass.initialize.apply(this,arguments);
 45 		if(this.targetEl){
 46 			for(var i=0,len=this.triggerEvents.length;i<len;i++){
 47 				this.targetEl.bind(this.triggerEvents[i],jQuery.proxy(this.showHandler,this));
 48 			}
 49 			
 50 			for(var i=0,len=this.hideEvents.length;i<len;i++){
 51 				this.targetEl.bind(this.hideEvents[i],jQuery.proxy(this.hideHandler,this));
 52 			}
 53 			
 54 			this.targetEl.bind("mousemove",jQuery.proxy(this.updateTipPos,this));
 55 		}	
 56 	},
 57 	createDom:function(){
 58 		Alipw.ToolTip.superclass.createDom.apply(this,arguments);
 59 	},
 60 	renderComplete:function(){
 61 		Alipw.ToolTip.superclass.renderComplete.apply(this,arguments);
 62 		
 63 		this.addEventListener("mouseover",this.showHandler,this,true);
 64 		this.addEventListener("mouseout",this.hideHandler,this,true);
 65 	},
 66 	showTip:function(){
 67 		this.show();
 68 		var x = this.lastPos[0];
 69 		var y = this.lastPos[1];
 70 
 71 		var windowSize = [Alipw.getWin().width(),Alipw.getWin().height()];
 72 		
 73 		var maxX = windowSize[0] + Alipw.getWin().scrollLeft() - 5;
 74 		var maxY = windowSize[1] + Alipw.getWin().scrollTop() - 5;
 75 		if(x + this.el.outerWidth() > maxX){
 76 			x = maxX - this.el.outerWidth();
 77 		}
 78 		
 79 		if(y + this.el.outerHeight() > maxY){
 80 			y = maxY - this.el.outerHeight();
 81 		}
 82 		
 83 		if(x < 0)x = 0;
 84 		if(y < 0)y = 0;
 85 		
 86 		this.setPosition(x,y);
 87 	},
 88 	_doLayout:function(){
 89 		if(this.textMaxWidth){
 90 			var body = this.getBody();
 91 			body.css("white-space","nowrap");
 92 			if(this.textMaxWidth < body.width()){
 93 				this.width = this.getWidthByBody(this.textMaxWidth);
 94 			}
 95 			body.css("white-space","");
 96 		}
 97 		
 98 		Alipw.ToolTip.superclass._doLayout.apply(this,arguments);
 99 	},
100 	//private
101 	showHandler:function(e){		
102 		clearTimeout(this.hideDelayTimer);
103 		clearTimeout(this.showDelayTimer);
104 		if(this.visible && this.rendered){
105 			return;
106 		}
107 		this.showDelayTimer = setTimeout(jQuery.proxy(this.showTip,this),this.showDelay * 1000);
108 	},
109 	//private
110 	hideHandler:function(e){
111 		clearTimeout(this.showDelayTimer);
112 		clearTimeout(this.hideDelayTimer);
113 		if(this.rendered){
114 			this.hideDelayTimer = setTimeout(jQuery.proxy(this.hide,this),this.hideDelay * 1000);
115 		}
116 	},
117 	//private
118 	updateTipPos:function(e){
119 		this.lastPos = [e.pageX + this.offsetX,e.pageY + this.offsetY];
120 	}
121 });
