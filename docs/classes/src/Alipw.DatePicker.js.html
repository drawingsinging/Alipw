135 				'<div class="{$baseCls}-navbar">',
136 					'<span class="{$baseCls}-navbar-info"></span>',
137 					'<a href="#" class="{$baseCls}-navbar-btn-prevyear"></a>',
138 					'<a href="#" class="{$baseCls}-navbar-btn-prevmonth"></a>',
139 					'<a href="#" class="{$baseCls}-navbar-btn-nextyear"></a>',

190 			this.updateNavigationInfo();
191 		},this));
192 		this.el.find('.' + this.baseCls + '-navbar-btn-nextyear').click(jQuery.proxy(function(e){
193 			e.preventDefault();
194 			var current = [this.date.getFullYear(),this.date.getMonth() + 1];
195 			var dateTo = [current[0] + 1,current[1]];
196 			this.setDate(new Date(dateTo[0],dateTo[1] - 1));

198 		},this));

200 			e.preventDefault();
201 			var current = [this.date.getFullYear(),this.date.getMonth() + 1];
202 			var dateTo = current[1] == 12 ? [current[0] + 1,1]:[current[0],current[1] + 1];
203 			this.setDate(new Date(dateTo[0],dateTo[1] - 1));

243 	renderCalendarHeader:function(){
244 		var html = '';
245 		html += '<tr>';
246 		for(var i=0;i<7;i++){
247 			html += '<th>' + this.weekText[i] + '</th>';
248 		}
249 		html += '</tr>';
250 		this.el.find('.'+ this.baseCls + '-calendar table thead').empty().append(html);
251 	},
252 	renderCells:function(year,month){
253 		var html = '';
254 		html += '<tr>';
255 		var cellData = this.getDays_DatePicker(year,month);
256 		this._calendarData = cellData;
257 		var cellCls;
258 		
259 		var today = new Date();
260 		todayYear = today.getFullYear();
261 		todayMonth = today.getMonth() + 1;
262 		todayDay = today.getDate();
263 		for(var i=0,len=cellData.length;i<len;i++){
264 			if(i != 0 && i % 7 == 0){
265 				html += '</tr><tr>';
266 			}
267 			if(cellData[i].isLastMonth){
268 				cellCls = this.baseCls + '-lastmonth';
269 			}else if(cellData[i].isNextMonth){
270 				cellCls = this.baseCls + '-nextmonth';
271 			}else if(cellData[i].year == todayYear && cellData[i].month == todayMonth && cellData[i].day == todayDay){
272 				cellCls = this.baseCls + '-today';
273 			}else{
274 				cellCls = '';
275 			}
276 			
277 			if(cellData[i].disabled){
278 				cellCls += ' ' + this.baseCls + '-cell-disabled';
279 			}
280 
281 			html += '<td class="' + cellCls + '" cellIndex="' + i + '" celldisabled="' + cellData[i].disabled.toString() + '">' + cellData[i].day.toString() + '</td>';
282 		}
283 		html += '</tr>';
284 		this.el.find('.'+ this.baseCls + '-calendar table tbody').empty().append(html);
285 	},
286 	/**
287 	 * @public

296 	getDays_DatePicker:function(year,month){
297 		var days = new Array();
298 		
299 		var getDayNum = function(year,month){
300 			var leap = (year%100==0 ? res=(year%400==0 ? 1 : 0) : res=(year%4==0 ? 1: 0));
301 			var daysNum = new Array(31,28+leap,31,30,31,30,31,31,30,31,30,31);
302 			return daysNum[month - 1];
303 		};
304 		
305 		//Zeller
306 		/*
307 		var c = parseInt(year / 100);
308 		var y = year - c * 100;
309 		y = month > 2 ? y : (y - 1);
310 		var m = month > 2 ? month : (month + 12);
311 		var d = 1;
312 		var w = y + parseInt( y /4 ) + parseInt( c / 4 ) - 2 * c + parseInt( 26 * ( m + 1 ) / 10 ) + d - 1;
313 		w = w % 7;
314 		if(w < 0)w = w + 7;
315 		*/
316 		
317 		var weekday = (new Date(year,month - 1)).getDay();
318 		var lastMonth = month == 1 ? 12 : (month - 1);
319 		var lastMonthYear =  month == 1 ? year - 1 : year;
320 		var lastMonthDaysInCurCalNum = weekday == 0 ? 7 : weekday;
321 		var lastMonthDayNum;
322 		
323 		if(lastMonthDaysInCurCalNum > 0){
324 			if(month == 1){
325 				lastMonthDayNum = getDayNum(year - 1,lastMonth);
326 			}else{
327 				lastMonthDayNum = getDayNum(year,lastMonth);
328 			}
329 		
330 		
331 			for(var i=lastMonthDayNum-lastMonthDaysInCurCalNum+1 ,len=lastMonthDayNum;i<=len;i++){
332 				days.push({
333 					year:lastMonthYear,
334 					month:lastMonth,
335 					day:i,
336 					isLastMonth:true,
337 					disabled:this.isDateDisabled_DatePicker(new Date(lastMonthYear + '/' + lastMonth + '/' + i))
338 				});
339 			}
340 		}
341 		
342 		for(var i=1,len=getDayNum(year,month);i<=len;i++){
343 			days.push({
344 				year:year,
345 				month:month,
346 				day:i,
347 				disabled:this.isDateDisabled_DatePicker(new Date(year + '/' + month + '/' + i))
348 			});
349 		}
350 		
351 		var restCellNum = 42 - days.length;
352 		var nextMonth = month == 12 ? 1 : (month + 1);
353 		var nextMonthYear =  month == 12 ? year + 1 : year;
354 		
355 		for(var i=1;i<=restCellNum;i++){
356 			days.push({
357 				year:nextMonthYear,
358 				month:nextMonth,
359 				day:i,
360 				isNextMonth:true,
361 				disabled:this.isDateDisabled_DatePicker(new Date(nextMonthYear + '/' + nextMonth + '/' + i))
362 			});
363 		}
364 		
365 		return days;
366 	},
367 	//private
368 	isDateDisabled_DatePicker:function(date){
369 		if(this.minDate && date && date < this.minDate){
370 			return true;
371 		}
372 		if(this.maxDate && date && date > this.maxDate){
373 			return true;
374 		}
375 		return false;
376 	},
377 	//private
378 	applyToElement_DatePicker:function(e){
379 		var input = jQuery(e.currentTarget);
380 		this.show();
381 		Alipw.ComponentManager.bringToFront(this);
382 		
383 		var winHeight = jQuery(window).height();
384 		var inputPos = input.offset();
385 		var height = this.el.outerHeight();
386 		var inputSize = [input.innerWidth(),input.innerHeight()];
387 		
388 		var left = inputPos.left;
389 		var top;
390 		
391 		if(winHeight - (inputPos.top - jQuery(document).scrollTop()) >= height){
392 			top = inputPos.top + inputSize[1];
393 		}else{
394 			top = inputPos.top - height;
395 		}
396 		this.setPosition(left,top);
397 	},
398 	//private
399 	documentClickHandler_DatePicker:function(e){
400 		if(Alipw.isInNode(e.target,this._applyToEl[0]) || e.target == this._applyToEl[0] || Alipw.isInNode(e.target,this.el[0]) || e.target == this.el[0]){
401 			return;
402 		}
403 		this.hide();
404 	}
