1 /**
  2  * @constructor Alipw.Panel
  3  * @extends Alipw.Container
  4  * @description 面板。面板是指一个含有标题栏，底部按钮栏的内容展示容器。
  5  * @example
  6  * 
  7  */
  8 
  9 Alipw.Panel = Alipw.extend(Alipw.Container,
 10 /** @lends Alipw.Panel.prototype */
 11 {
 12 	baseCls:"alipw-panel",
 13 	showHeader:true,
 14 	closable:false,
 15 	closeAction:"remove",
 16 	closeBtnConfig:{},
 17 	minWidth:200,
 18 	minHeight:150,
 19 	width:200,
 20 	height:150,
 21 	title:"",
 22 	buttons:"",
 23 	buttonAlign:"center",
 24 	buttonGap:6,
 25 	constructor:function(){
 26 		Alipw.Panel.superclass.constructor.apply(this,arguments);
 27 	},
 28 	commitProperties:function(){
 29 		Alipw.Panel.superclass.commitProperties.apply(this,arguments);
 30 	},
 31 	initialize:function(){
 32 		Alipw.Panel.superclass.initialize.apply(this,arguments);
 33 		this.headerButtonItems = new Array();
 34 		this.buttonItems = new Array();
 35 	},
 36 	createDom:function(){
 37 		Alipw.Panel.superclass.createDom.apply(this,arguments);
 38 		this.el.append([
 39 				    '<div class="' + this.baseCls +'-wrap">',
 40 						'<div class="' + this.baseCls +'-header">',
 41 					    	'<span class="' + this.baseCls +'-header-left">',
 42 					        	'<span class="' + this.baseCls +'-header-right">',
 43 					            	'<span class="' + this.baseCls +'-header-center">',
 44 					                	'<span class="' + this.baseCls +'-header-text">',
 45 					                    	'<span class="' + this.baseCls +'-header-text-shadow"></span>',
 46 					                        '<span class="' + this.baseCls +'-header-text-fore"></span>',
 47 					                    '</span>',
 48 					                    '<div class="' + this.baseCls +'-header-buttons"></div>',
 49 					                '</span>',
 50 					            '</span>',
 51 					        '</span>',
 52 					    '</div>',
 53 					    '<div class="' + this.baseCls +'-body-wrap">',
 54 					    	'<div class="' + this.baseCls +'-body-left">',
 55 					    		'<div class="' + this.baseCls +'-body-right">',
 56 					    			'<div class="' + this.baseCls +'-body">',
 57 					    				'<div class="' + this.baseCls +'-body-content">',
 58 					    				'</div>',
 59 						    			'<div class="' + this.baseCls +'-body-buttonbar">',
 60 									    '</div>',
 61 					    			'</div>',
 62 					    		'</div>',
 63 					    	'</div>',
 64 					    '</div>',
 65 					    '<div class="' + this.baseCls +'-footer">',
 66 						    '<span class="' + this.baseCls +'-footer-left">',
 67 					    		'<span class="' + this.baseCls +'-footer-right">',
 68 					    			'<span class="' + this.baseCls +'-footer-center"></span>',
 69 					    		'</span>',
 70 					    	'</span>',
 71 					    '</div>',
 72 				    '</div>'
 73 				].join(""));
 74 		
 75 		this.header = this.el.find("." + this.baseCls +"-header");
 76 		this.body = this.getBody();
 77 		this.footer = this.el.find("." + this.baseCls +"-footer");
 78 		this.buttonBar = this.el.find("." + this.baseCls +"-body-buttonbar");
 79 		this.bodyWrap = this.el.find("." + this.baseCls +"-body-wrap");
 80 		
 81 		this.addEventListener('destroy',function(){
 82 			this.headerButtonItems = null;
 83 			this.buttonItems = null;
 84 		},this);
 85 	},
 86 	renderComplete:function(){
 87 		Alipw.Panel.superclass.renderComplete.apply(this,arguments);
 88 
 89 		this.setTitle(this.title);
 90 		this.renderHeaderBtns();
 91 		
 92 		if(this.buttons instanceof Array && this.buttons.length > 0){
 93 			this.addButtons(this.buttons);
 94 		}
 95 	},
 96 	//protected
 97 	_doLayout:function(){
 98 		Alipw.Panel.superclass._doLayout.apply(this,arguments);
 99 		
100 		var wrap = this.el.find("." + this.baseCls + "-wrap");
101 		if(this.showHeader){
102 			wrap.removeClass(this.baseCls + "-noheader");
103 		}else{
104 			wrap.addClass(this.baseCls + "-noheader");
105 		}
106 		
107 		if(this.buttonItems.length > 0){
108 			this.buttonBar.css("text-align",this.buttonAlign);
109 			this.buttonBar.show();
110 		}else{
111 			this.buttonBar.hide();
112 		}
113 		
114 		var wrap = this.el.find("." + this.baseCls + "-wrap");
115 		var wrapHeight = wrap.outerHeight();
116 		var wrapWidth = wrap.outerWidth();
117 		
118 		if(this.height){
119 			//this.body.height(this.height - (this.showHeader?this.header.outerHeight():0) - (hasButton?this.buttonBar.outerHeight():0) - (this.bodyWrap.outerHeight() - this.body.parent().height()) - this.footer.height());
120 			this.body.height(this.height - (wrapHeight - this.body.height()));
121 		}
122 		
123 		if(this.width){
124 			this.body.width('auto');
125 			this.body.width(this.width - (wrapWidth - this.body.width()));
126 		}
127 	},
128 	setTitle:function(title){
129 		this.el.find("." + this.baseCls + "-header-text-shadow").html(title);
130 		this.el.find("." + this.baseCls + "-header-text-fore").html(title);
131 	},
132 	//protected
133 	renderHeaderBtns:function(){
134 		if(this.closable){
135 			if(!this.closeBtn){
136 				this.closeBtn = new Alipw.Button(jQuery.extend({
137 					cls:"button-alipw-panel-header-red alipw-panel-btn-close",
138 					iconCls:"icon-alipw-panel-closeBtn",
139 					label:"",
140 					width:45,
141 					height:22,
142 					inlineStyle:{"margin-right":"-1px"},
143 					renderTo:this.el.find("." + this.baseCls + "-header-buttons"),
144 					handler:this.closeBtnHandler,
145 					scope:this
146 				},this.closeBtnConfig));
147 				this.headerButtonItems.push(this.closeBtn);
148 				
149 				this.closeBtn.addEventListener('mousedown',function(e){
150 					e.preventDefault();
151 					e.stopPropagation();
152 				},this,true);
153 			}
154 		}
155 
156 		if(this.headerButtonItems.length < 1){
157 			this.el.find("." + this.baseCls + "-header-buttons").hide();
158 		}else{
159 			this.el.find("." + this.baseCls + "-header-buttons").show();
160 		}
161 	},
162 	close:function(){
163 		this.fireEvent('beforeClose',{},false);
164 		if(this.closeAction == "remove"){
165 			this.destroy();
166 		}else if(this.closeAction = "hide"){
167 			this.setVisible(false);
168 		}
169 	},
170 	getWidthByBody:function(width){
171 		return width + (this.getWidth() - this.body.width());
172 	},
173 	getHeightByBody:function(height){
174 		return height + (this.getHeight() - this.body.height());
175 	},
176 	addButtons:function(config){
177 		if(config instanceof Array){
178 			for(var i=0,len=config.length;i<len;i++){
179 				this.createButton(config[i]);
180 			}
181 			this.doLayout();
182 		}else if(typeof(config) == "object"){
183 			var button = this.createButton(config);
184 			this.doLayout();
185 			return button;
186 		}
187 	},
188 	getButton:function(id){
189 		
190 	},
191 	//protected
192 	getBody:function(){
193 		return this.el.find("." + this.baseCls + "-body-content");
194 	},
195 	closeBtnHandler:function(e){
196 		e.stopPropagation();
197 		this.close();
198 	},
199 	//private
200 	createButton:function(config){
201 		var button = new Alipw.Button(jQuery.extend({
202 			renderTo:this.buttonBar,
203 			width:60,
204 			inlineStyle:{
205 				margin:"0 " + this.buttonGap/2 + "px"
206 			}
207 		},config));
208 		this.buttonItems.push(button);
209 		button.ownerComp = this;
210 		
211 		return button;
212 	},
213 	//private
214 	createButtons_Panel:function(){
215 		this.addButtons(this.buttons);
216 	}
217 });
