1 /** 
  2 * @class
  3 * @type singleton
  4 * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5 * @description 消息提示器。
  6 */
  7 Alipw.Msg = function(){
  8 	
  9 	var createMsgWindow = function(title,msg,fn,windowConfig,msgConfig,buttonsConfig){
 10 		var defaultConfig = {
 11 			title:"Message",
 12 			buttonAlign:"right"
 13 		};
 14 		
 15 		var html = new Alipw.Template([
 16             '<div class="alipw-msg">',
 17             	'<div class="alipw-msg-text">',
 18            			'{$msg}',
 19            			'<div class="alipw-msg-icon {$iconCls}"></div>',
 20        			'</div>',
 21    			'</div>'
 22        	]).set({
 23    			msg:msg,
 24    			iconCls:msgConfig.iconCls
 25         }).html;
 26 		
 27 		var winConfig = jQuery.extend(defaultConfig, windowConfig, {
 28 			html:html,
 29 			title:title,
 30 			resizable:false,
 31 			autoCenter:true,
 32 			autoRender:false
 33 		});
 34 		
 35 		var win = new Alipw.Window(winConfig);
 36 		win.setVisible(false,true);
 37 		win.render();
 38 		
 39 		if(buttonsConfig instanceof Array){
 40 			for(var i=0,len=buttonsConfig.length;i<len;i++){
 41 				var button = win.addButtons(buttonsConfig[i]);
 42 				button.addEventListener("click",function(e){
 43 					if(typeof(fn) == "function"){
 44 						fn(this._callbackToken);
 45 					}
 46 					win.close();
 47 				},button,true);
 48 			}
 49 		}
 50 		
 51 		var msgBody = win.el.find(".alipw-msg");
 52 		msgBody.css("white-space","nowrap");
 53 		win.body.css("overflow","hidden");
 54 		if(msgBody.width() > msgConfig.textMaxWidth){
 55 			msgBody.width(msgConfig.textMaxWidth);
 56 			msgBody.css("white-space","");
 57 		}else if(msgBody.width() < msgConfig.textMinWidth){
 58 			msgBody.width(msgConfig.textMinWidth);
 59 			msgBody.css("white-space","");
 60 		}
 61 		
 62 		var winWidth = win.getWidthByBody(msgBody.outerWidth());
 63 		var winHeight = win.getHeightByBody(msgBody.outerHeight());
 64 		
 65 		win.setSize(winWidth,winHeight);
 66 		win.center();
 67 		Alipw.ComponentManager.bringToFront(win);
 68 		win.setVisible(true);
 69 		
 70 		return win;
 71 	};
 72 
 73 	var pub = 
 74 	/** @lends Alipw.Msg */
 75 	{
 76 		/**
 77 		 * @public
 78 		 * @description 警告提示窗。
 79 		 * @param {String} title 提示窗标题。
 80 		 * @param {String} msg 提示窗显示内容
 81 		 * @param {Function} [fn] 点击确定后响应的function。
 82 		 * @param {Object} [config] 配置选项以及额外的Alipw.Window配置选项。除Alipw.Window的配置选项外，还有一下可设置属性：<ul><li>iconCls : String  提示窗中icon的class名称。默认为'alipw-icon-msg-warning'</li><li>OKText : String  确定按钮的显示文本。默认为'OK'</li><li>textMaxWidth : Number 显示的最大宽度。若超过此宽度，则文本将折行显示。默认为500</li><li>textMinWidth : Number 显示的最小宽度。默认为150</li></ul>
 83 		 */
 84 		alert:function(title,msg,fn,config){
 85 			if(typeof arguments[0] == 'object'){
 86 				var	args	= arguments[0], 
 87 					title	= args.title || '提醒',
 88 					msg		= args.msg,
 89 					fn		= args.fn,
 90 					config	= args.config;
 91 			}
 92 			if(typeof(config) != "object"){
 93 				config = new Object();
 94 			}
 95 			
 96 			var msgConfig = {
 97 				OKText:config.OKText || "OK",
 98 				iconCls:config.iconCls || "alipw-icon-msg-warning",
 99 				textMaxWidth:config.textMaxWidth || 500,
100 				textMinWidth:config.textMinWidth || 150
101 			};
102 			
103 			for(var i in msgConfig){
104 				delete config[i];
105 			}
106 			
107 			config = jQuery.extend({
108 				modal:true
109 			},config);
110 			
111 			var buttonsConfig = [{
112 				label:msgConfig.OKText,
113 				_callbackToken:"OK"
114 			}];
115 			
116 			var win = createMsgWindow(title,msg,fn,config,msgConfig,buttonsConfig);
117 			return win;
118 		},
119 		prompt:function(title,msg,fn,config){
120 			
121 		},
122 		/**
123 		 * @public
124 		 * @description 确认提示窗。
125 		 * @param {String} title 提示窗标题。
126 		 * @param {String} msg 提示窗显示内容
127 		 * @param {Function} [fn] 点击按钮后响应的function。function会被传入一个action参数，action的值为'yes'或'no'。
128 		 * @param {Object} [config] 配置选项以及额外的Alipw.Window配置选项。除Alipw.Window的配置选项外，还有一下可设置属性：<ul><li>iconCls : String  提示窗中icon的class名称。默认为'alipw-icon-msg-question'</li><li>yesText : String  '是'按钮的显示文本。默认为'Yes'</li><li>noText : String  '否'按钮的显示文本。默认为'No'</li><li>textMaxWidth : Number 显示的最大宽度。若超过此宽度，则文本将折行显示。默认为500</li><li>textMinWidth : Number 显示的最小宽度。默认为150</li></ul>
129 		 */
130 		confirm:function(title,msg,fn,config){
131 			if(typeof(config) != "object"){
132 				config = new Object();
133 			}
134 			
135 			var msgConfig = {
136 				yesText:config.yesText || "Yes",
137 				noText:config.noText || "No",
138 				iconCls:config.iconCls || "alipw-icon-msg-question",
139 				textMaxWidth:config.textMaxWidth || 500,
140 				textMinWidth:config.textMinWidth || 150
141 			};
142 			
143 			for(var i in msgConfig){
144 				delete config[i];
145 			}
146 			
147 			config = jQuery.extend({
148 				modal:true
149 			},config);
150 			
151 			var buttonsConfig = [{
152 				label:msgConfig.yesText,
153 				_callbackToken:"yes"
154 			},{
155 				label:msgConfig.noText,
156 				_callbackToken:"no"
157 			}];
158 			
159 			var win = createMsgWindow(title,msg,fn,config,msgConfig,buttonsConfig);
160 			return win;
161 		},
162 		tip:function(msg,config){
163 			if(typeof(config) != "object"){
164 				config = new Object();
165 			}
166 			
167 			var msgConfig = {
168 				iconCls:config.iconCls || "alipw-icon-msg-success",
169 				textMaxWidth:config.textMaxWidth || 500,
170 				textMinWidth:config.textMinWidth || 150,
171 				timeout:function(){
172 					if(config.timeout === 0){
173 						return 100000000000000;
174 					}else if(!config.timeout){
175 						return 2000;
176 					}else{
177 						return config.timeout;
178 					}
179 				}()
180 			};
181 			
182 			for(var i in msgConfig){
183 				delete config[i];
184 			}
185 			
186 			config = jQuery.extend({
187 				showHeader:false,
188 				modal:false,
189 				draggable:false
190 			},config);
191 			
192 			var win = createMsgWindow("",msg,null,config,msgConfig);
193 			setTimeout(function(){
194 				win.close();
195 			},msgConfig.timeout);
196 			
197 			return win;
198 		}
199 	};
200 	
201 	return pub;
202 }();
