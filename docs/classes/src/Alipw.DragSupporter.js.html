1 /**
  2  * @constructor Alipw.DragSupporter
  3  * @extends Alipw.Nonvisual
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 拖拽支持类，可应用于可视控件，使其可被鼠标拖拽。
  6  */
  7 
  8 Alipw.DragSupporter = Alipw.extend(Alipw.Nonvisual,
  9 /** @lends Alipw.DragSupporter.prototype */
 10 {
 11 	/**
 12 	 * @property
 13 	 * @type HTML Element/Selector/Alipw.Component
 14 	 * @description [config option]指定需要支持拖拽的目标。
 15 	 * @default ''
 16 	 */
 17 	target:"",
 18 	/**
 19 	 * @property
 20 	 * @type HTML Element/Selector
 21 	 * @description [config option]指定鼠标的拖拽点。
 22 	 * @default ''
 23 	 */
 24 	dragEl:"",
 25 	/**
 26 	 * @property
 27 	 * @type Boolean
 28 	 * @description [config option]定义拖拽时是否使用代理显示对象。
 29 	 * @default true
 30 	 */
 31 	useProxy:true,
 32 	/**
 33 	 * @property
 34 	 * @type Boolean
 35 	 * @description [config option]定义是否支持拖拽动作的响应。
 36 	 * @default true
 37 	 */
 38 	draggable:true,
 39 	/**
 40 	 * @property
 41 	 * @type Boolean
 42 	 * @description [config option]定义是否支持放下动作的响应。
 43 	 * @default false
 44 	 */
 45 	droppable:false,
 46 	/**
 47 	 * @property
 48 	 * @type Boolean
 49 	 * @description [config option]定义拖拽时拖拽对象的class名称。
 50 	 * @default 'alipw-component-dragging'
 51 	 */
 52 	draggingCls:"alipw-component-dragging",
 53 	/**
 54 	 * @property
 55 	 * @type Boolean
 56 	 * @description [config option]定义拖是否可用。
 57 	 * @default true
 58 	 */
 59 	enabled:true,
 60 	constructor:function(){
 61 		Alipw.DragSupporter.superclass.constructor.apply(this,arguments);
 62 	},
 63 	commitProperties:function(){
 64 		Alipw.DragSupporter.superclass.commitProperties.apply(this,arguments);
 65 		
 66 		if(this.target instanceof Alipw.Component){
 67 			this.targetEl = this.target.el;
 68 		}else{
 69 			this.targetEl = Alipw.convertEl(this.target);
 70 		}
 71 		
 72 	},
 73 	initialize:function(){
 74 		Alipw.DragSupporter.superclass.initialize.apply(this,arguments);
 75 		
 76 		if(this.targetEl[0]){
 77 			var dragEl = this.dragEl?jQuery(this.dragEl):this.targetEl;
 78 			
 79 			dragEl.bind("mousedown",jQuery.proxy(this.getReady,this));
 80 			jQuery(document).bind("mousemove",jQuery.proxy(this.move,this));
 81 			jQuery(document).bind("mouseup",jQuery.proxy(this.release,this));
 82 		}
 83 	},
 84 	//private
 85 	getReady:function(e){
 86 		if(!this.enabled)return;
 87 		
 88 		e.preventDefault();
 89 		this.ready = true;
 90 		this._dx = this.targetEl.offset().left - e.pageX;
 91 		this._dy = this.targetEl.offset().top - e.pageY;
 92 	},
 93 	//private
 94 	move:function(e){
 95 		if(!this.enabled || !this.ready)return;
 96 		e.preventDefault();
 97 
 98 		if(!this.dragging){
 99 			this.dragging = true;
100 			
101 			if(this.useProxy){
102 				if(!this.proxy){
103 					this.proxy = new Alipw.BorderContainer({
104 						cls:"alipw-dragproxy",
105 						floating:true,
106 						position:(this.target instanceof Alipw.Component)?this.target.position:""
107 					});
108 					this.addEventListener('destroy',function(){
109 						if(this.proxy){
110 							this.proxy.destroy();
111 							this.proxy = null;
112 						}
113 					},this);
114 				}
115 				this.proxy.show();
116 				this.proxy.setWidth(this.targetEl.width());
117 				this.proxy.setHeight(this.targetEl.height());
118 				Alipw.ComponentManager.bringToFront(this.proxy);
119 			}else{
120 				if(this.draggingCls){
121 					this.targetEl.addClass(this.draggingCls);
122 				}
123 			}
124 			
125 			if(this.target instanceof Alipw.Component){
126 				this.originalPosition = [this.target.getX(),this.target.getY()];
127 				this.target.fireEvent("dragstart",{},false);
128 			}else{
129 				this.originalPosition = [this.targetEl.offset().left,this.targetEl.offset().top];
130 				this.targetEl.triggerHandler("alipw-dragstart");
131 			}
132 		}
133 		if(this.target instanceof Alipw.Component){
134 			if(this.target.position == "fixed"){
135 				this._currCoord = this.getHandledPosition_DragSupporter(e.clientX + this._dx, e.clientY + this._dy);
136 				(this.proxy || this.target).setPosition(this._currCoord[0],this._currCoord[1],null,false);
137 			}else{
138 				this._currCoord = this.getHandledPosition_DragSupporter(e.pageX + this._dx,e.pageY + this._dy);
139 				(this.proxy || this.target).el.offset({
140 					left:this._currCoord[0],
141 					top:this._currCoord[1]
142 				});
143 				//this.target.setPosition(e.pageX + this._dx,e.pageY + this._dy);
144 			}
145 			this.target.fireEvent("drag",{
146 				left:this._currCoord[0],
147 				top:this._currCoord[1]
148 			},false);
149 		}else{
150 			this._currCoord = this.getHandledPosition_DragSupporter(e.pageX + this._dx,e.pageY + this._dy);
151 			if(this.proxy){
152 				this.proxy.el.offset({
153 					left:this._currCoord[0],
154 					top:this._currCoord[1]
155 				});
156 			}else{
157 				this.targetEl.offset({
158 					left:this._currCoord[0],
159 					top:this._currCoord[1]
160 				});
161 			}
162 			
163 			this.targetEl.triggerHandler("alipw-drag",{
164 				left:this._currCoord[0],
165 				top:this._currCoord[1]
166 			});
167 		}
168 	},
169 	//private
170 	release:function(e){
171 		if(!this.enabled)return;
172 		
173 		if(this.dragging){
174 			this.dragging = false;
175 			if(this.target instanceof Alipw.Component){
176 				this.target.setPosition(this.originalPosition[0],this.originalPosition[1],null,false);
177 				
178 				this.target.fireEvent("dragend",{
179 					left:this._currCoord[0],
180 					top:this._currCoord[1]
181 				},false);
182 			}else{
183 				this.targetEl.offset({
184 					left:this.originalPosition[0],
185 					top:this.originalPosition[1]
186 				});
187 				this.targetEl.triggerHandler("alipw-dragend",{
188 					left:this._currCoord[0],
189 					top:this._currCoord[1]
190 				});
191 			}
192 		}
193 		if(this.draggingCls){
194 			this.targetEl.removeClass(this.draggingCls);
195 		}
196 		if(this.proxy){
197 			this.proxy.setVisible(false);
198 		}
199 		this.dragging = false;
200 		this.ready = false;
201 	},
202 	//private
203 	getHandledPosition_DragSupporter:function(left,top){
204 		var jWin = jQuery(window);
205 		var winWidth = jWin.width();
206 		var winHeight = jWin.height();
207 		var docWidth = Alipw.getDocWidth();
208 		var docHeight = Alipw.getDocHeight();
209 		var targetWidth,targetHeight; 
210 		targetWidth = this.targetEl.outerWidth();
211 		targetHeight = this.targetEl.outerHeight();
212 
213 		
214 		if(this.target.position == "fixed"){
215 			
216 			if(left + targetWidth > winWidth - 10){
217 				left = winWidth - targetWidth - 10;
218 			}
219 			
220 			if(left < 10){
221 				left = 10;
222 			}
223 			
224 			if(top + targetHeight > winHeight - 10){
225 				top = winHeight - targetHeight - 10;
226 			}
227 			
228 			if(top < 10){
229 				top = 10;
230 			}
231 		}else{
232 			var conWidth = winWidth > docWidth? winWidth : docWidth;
233 			var conHeight = winHeight > docHeight? winHeight : docHeight;
234 			
235 			if(left + targetWidth > conWidth - 10){
236 				left = conWidth - targetWidth - 10;
237 			}
238 			
239 			if(left < 10){
240 				left = 10;
241 			}			
242 			
243 			if(top + targetHeight > conHeight - 10){
244 				top = conHeight - targetHeight - 10;
245 			}
246 			
247 			if(top < 10){
248 				top = 10;
249 			}
250 		}
251 		
252 		return [left,top];
253 	}
254 });
