1 /**
  2  * @class Alipw.Config
  3  * @type singleton
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 用于设定Alipw中所有类的默认配置。
  6  */
  7 
  8 Alipw.Config = function(){
  9 	var configs = new Object();
 10 	
 11 	var pub = {
 12 		set:function(className, config){
 13 			if(!configs[className]){
 14 				configs[className] = new Object();
 15 			}
 16 			
 17 			Alipw.apply(configs[className],config);
 18 			
 19 			if(Alipw.mode == 'all'){
 20 				var classObject = Alipw.getObjectByName(className);
 21 				if(classObject){
 22 					if(classObject.prototype){
 23 						Alipw.apply(classObject.prototype,configs[className]);
 24 					}else{
 25 						Alipw.apply(classObject.prototype,configs[className]);
 26 					}
 27 				}
 28 			}
 29 		},
 30 		get:function(className){
 31 			return configs[className];
 32 		}
 33 	};
 34 
 35 	return pub;
 36 }();
