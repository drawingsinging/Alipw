1 /**
  2  * @constructor Alipw.Shadow
  3  * @extends Alipw.BoxComponent
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 阴影。应用于盒状的可视控件使其具有阴影效果。
  6  * @example
  7  * 
  8  */
  9 
 10 Alipw.Shadow = Alipw.extend(Alipw.BoxComponent,
 11 /** @lends Alipw.Shadow.prototype */
 12 {
 13 	floating:true,
 14 	/*
 15 	 * sides : Shadow displays on both sides and bottom only
 16  	 * frame : Shadow displays equally on all four sides
 17 	 * drop : Traditional bottom-right drop shadow
 18 	 * */
 19 	mode:"drop",
 20 	target:null,
 21 	floatingManagement:false,
 22 	baseCls:"alipw-shadow",
 23 	constructor:function(config){
 24 		Alipw.Shadow.superclass.constructor.apply(this,arguments);
 25 	},
 26 	commitProperties:function(){
 27 		Alipw.Shadow.superclass.commitProperties.apply(this,arguments);
 28 	},
 29 	initialize:function(){
 30 		Alipw.Shadow.superclass.initialize.apply(this,arguments);		
 31 	},
 32 	createDom:function(){
 33 		Alipw.Shadow.superclass.createDom.apply(this,arguments);
 34 		
 35 		if(Alipw.ie() && parseInt(Alipw.ie())<7){
 36 			this.el.attr("class","alipw-ie-shadow");
 37 		}else{
 38 			this.el.append([
 39 				'<div class="xst">',
 40 				'<div class="xstl"></div>',
 41 				'<div class="xstc"></div>',
 42 				'<div class="xstr"></div>',
 43 				'</div>',
 44 				'<div class="xsc">',
 45 				'<div class="xsml"></div>',
 46 				'<div class="xsmc"></div>',
 47 				'<div class="xsmr"></div>',
 48 				'</div>',
 49 				'<div class="xsb">',
 50 				'<div class="xsbl"></div>',
 51 				'<div class="xsbc"></div>',
 52 				'<div class="xsbr"></div>',
 53 				'</div>'
 54 			].join(""));
 55 		}
 56 		
 57 		if(this.target instanceof Alipw.BoxComponent){
 58 			var applyToElement = Alipw.createFuncProxy(this.applyToElement,this);
 59 			
 60 			this.target.addEventListener("resize",applyToElement);
 61 			this.target.addEventListener("move",applyToElement);
 62 			this.target.addEventListener("visibilityChange",applyToElement);
 63 			this.target.addEventListener("afterRemove",applyToElement);
 64 			this.target.addEventListener("afterRender",applyToElement);
 65 			
 66 			this.addEventListener('destroy',function(){
 67 				this.target.removeEventListener("resize",applyToElement);
 68 				this.target.removeEventListener("move",applyToElement);
 69 				this.target.removeEventListener("visibilityChange",applyToElement);
 70 				this.target.removeEventListener("afterRemove",applyToElement);
 71 				this.target.removeEventListener("afterRender",applyToElement);
 72 			},this);
 73 		}
 74 	},
 75 	render:function(o){
 76 		Alipw.Shadow.superclass.render.apply(this,arguments);
 77 		if(this.target){
 78 			this.applyToElement();
 79 		}
 80 	},
 81 	renderComplete:function(){
 82 		Alipw.Shadow.superclass.renderComplete.apply(this,arguments);
 83 	},
 84 	//protected
 85 	_doLayout:function(){
 86 		Alipw.Shadow.superclass._doLayout.apply(this,arguments);
 87 		if(!(Alipw.ie() && parseInt(Alipw.ie())<7)){
 88 			if(this.width){
 89 				if(this.width<12){
 90 					this.width = 12;
 91 				}
 92 				this.el.find(".xstc").width(this.width - 12);
 93 				this.el.find(".xsmc").width(this.width - 12);
 94 				this.el.find(".xsbc").width(this.width - 12);
 95 			}
 96 			if(this.height){
 97 				if(this.height<12){
 98 					this.height = 12;
 99 				}
100 				this.el.find(".xsc").height(this.height - 12);
101 			}
102 		}else{
103 			this.el.width(this.width - 9);
104 			this.el.height(this.height - 9);
105 		}
106 	},
107 	enable:function(){
108 		Alipw.Shadow.superclass.enable.apply(this,arguments);
109 		this.setVisible(true);
110 		this.applyToElement();
111 	},
112 	disable:function(){
113 		Alipw.Shadow.superclass.disable.apply(this,arguments);
114 		this.setVisible(false);
115 	},
116 	applyToElement:function(e){
117 		var o = this.target;
118 		//var shadowHorEdge = parseInt(Alipw.ie())<7?0:5;
119 		//var shadowVerEdge = parseInt(Alipw.ie())<7?0:4;
120 		if(o instanceof Alipw.BoxComponent && o.floating){
121 			var deltaX,deltaY,deltaW,deltaH;
122 			if(this.mode == "drop"){
123 				deltaX = deltaY = -3;
124 				deltaW = deltaH = 8; 
125 			}else if(this.mode == "frame"){
126 				deltaX = deltaY = -5;
127 				deltaW = deltaH = 10; 
128 			}else if(this.mode == "sides"){
129 				deltaX = -5;
130 				deltaY = -3;
131 				deltaW = 10;
132 				deltaH = 8;
133 			}
134 			
135 			this.setWidth(o.getWidth() + deltaW);
136 			this.setHeight(o.getHeight() + deltaH);
137 			
138 			this.setPosition(o.getX() + deltaX, o.getY() + deltaY, o.getZIndex() - 1, true, false);
139 			
140 			if(this.enabled){
141 				var visible;
142 				if(o.removed){
143 					visible = false;
144 				}else{
145 					visible = o.visible;
146 				}
147 				this.setVisible(visible);
148 			}else{
149 				this.setVisible(false);
150 			}
151 		}
152 	}
153 });
