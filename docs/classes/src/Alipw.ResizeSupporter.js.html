1 /**
  2  * @constructor Alipw.ResizeSupporter
  3  * @extends Alipw.Nonvisual
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 缩放支持类。该类可应用与盒状可视控件(Alipw.BoxComponent)使其能够通过鼠标拖拽来改变尺寸。
  6  */
  7 
  8 Alipw.ResizeSupporter = Alipw.extend(Alipw.Nonvisual,
  9 /** @lends Alipw.ResizeSupporter.prototype */
 10 {
 11 	target:"",
 12 	minWidth:16,
 13 	minHeight:16,
 14 	maxWidth:"",
 15 	maxHeight:"",
 16 	constructor:function(){
 17 		Alipw.ResizeSupporter.superclass.constructor.apply(this,arguments);
 18 	},
 19 	commitProperties:function(){
 20 		Alipw.ResizeSupporter.superclass.commitProperties.apply(this,arguments);
 21 		
 22 		if(this.target instanceof Alipw.Component){
 23 			this.targetEl = this.target.el;
 24 		}else{
 25 			this.targetEl = Alipw.convertEl(this.target);
 26 		}
 27 	},
 28 	initialize:function(){
 29 		Alipw.ResizeSupporter.superclass.initialize.apply(this,arguments);
 30 		
 31 		if(this.targetEl.css("position") == "static"){
 32 			this.targetEl.addClass("alipw-resizesupporter");
 33 		}
 34 
 35 		this.topLeftCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-top-left"></div>');
 36 		this.topCenterCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-top-center"></div>');
 37 		this.topRightCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-top-right"></div>');
 38 		this.middleLeftCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-middle-left"></div>');
 39 		this.middleRightCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-middle-right"></div>');
 40 		this.bottomLeftCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-bottom-left"></div>');
 41 		this.bottomCenterCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-bottom-center"></div>');
 42 		this.bottomRightCorner = jQuery('<div class="alipw-resizesupporter-edge alipw-resizesupporter-edge-bottom-right"></div>');
 43 		
 44 		this.targetEl
 45 			.append(this.topCenterCorner)
 46 			.append(this.middleLeftCorner)
 47 			.append(this.middleRightCorner)
 48 			.append(this.bottomCenterCorner)
 49 			.append(this.topLeftCorner)
 50 			.append(this.topRightCorner)
 51 			.append(this.bottomLeftCorner)
 52 			.append(this.bottomRightCorner);
 53 		
 54 		this.topLeftCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 55 		this.topCenterCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 56 		this.topRightCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 57 		this.middleLeftCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 58 		this.middleRightCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 59 		this.bottomLeftCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 60 		this.bottomCenterCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 61 		this.bottomRightCorner.bind("mousedown",jQuery.proxy(this.getReady_ResizeSupporter,this));
 62 		
 63 		//create non-prototype-common functions to bind document
 64 		var moveHandler = Alipw.createFuncProxy(this.move_ResizeSupporter,this);
 65 		var releaseHandler = Alipw.createFuncProxy(this.release_ResizeSupporter,this);
 66 
 67 		jQuery(document).bind("mousemove",moveHandler);
 68 		jQuery(document).bind("mouseup",releaseHandler);
 69 		this.addEventListener('destroy',function(){
 70 			jQuery(document).unbind("mousemove",moveHandler);
 71 			jQuery(document).unbind("mouseup",releaseHandler);
 72 		},this);
 73 		
 74 		//for the damn ie 6!
 75 		if(parseInt(Alipw.ie()) < 7){
 76 			var _proxyResize4IE6 = function (){
 77 				this.middleRightCorner.height(this.targetEl.height());
 78 				this.middleLeftCorner.height(this.targetEl.height());
 79 				this.topCenterCorner.width(this.targetEl.width());
 80 				this.bottomCenterCorner.width(this.targetEl.width());
 81 			};
 82 			
 83 			_proxyResize4IE6.call(this);
 84 			this.targetEl.bind("resize",jQuery.proxy(_proxyResize4IE6,this));
 85 			this.addEventListener('destroy',function(){
 86 				this.targetEl.unbind("resize",_proxyResize4IE6);
 87 			},this);
 88 		}
 89 	},
 90 	enable:function(){
 91 		Alipw.ResizeSupporter.superclass.enable.apply(this,arguments);
 92 		this.topLeftCorner.show();
 93 		this.topCenterCorner.show();
 94 		this.topRightCorner.show();
 95 		this.middleLeftCorner.show();
 96 		this.middleRightCorner.show();
 97 		this.bottomLeftCorner.show();
 98 		this.bottomCenterCorner.show();
 99 		this.bottomRightCorner.show();
100 	},
101 	disable:function(){
102 		Alipw.ResizeSupporter.superclass.disable.apply(this,arguments);
103 		this.topLeftCorner.hide();
104 		this.topCenterCorner.hide();
105 		this.topRightCorner.hide();
106 		this.middleLeftCorner.hide();
107 		this.middleRightCorner.hide();
108 		this.bottomLeftCorner.hide();
109 		this.bottomCenterCorner.hide();
110 		this.bottomRightCorner.hide();
111 	},
112 	//private
113 	getReady_ResizeSupporter:function(e){
114 		if(!this.enabled)return;
115 		
116 		e.preventDefault();
117 		e.stopPropagation();
118 		
119 		if(e.currentTarget.setCapture){
120 			e.currentTarget.setCapture();
121 		}else if(window.captureEvents){  
122 			window.captureEvents(Event.MOUSEMOVE|Event.MOUSEUP);
123 		}
124 		this.ready = true;
125 		
126 		if(e.currentTarget.className.indexOf("edge-top-left") != -1){
127 			this.type = "nw";
128 		}else if(e.currentTarget.className.indexOf("edge-top-center") != -1){
129 			this.type = "n";
130 		}else if(e.currentTarget.className.indexOf("edge-top-right") != -1){
131 			this.type = "ne";
132 		}else if(e.currentTarget.className.indexOf("edge-middle-left") != -1){
133 			this.type = "w";
134 		}else if(e.currentTarget.className.indexOf("edge-middle-right") != -1){
135 			this.type = "e";
136 		}else if(e.currentTarget.className.indexOf("edge-bottom-left") != -1){
137 			this.type = "sw";
138 		}else if(e.currentTarget.className.indexOf("edge-bottom-center") != -1){
139 			this.type = "s";
140 		}else if(e.currentTarget.className.indexOf("edge-bottom-right") != -1){
141 			this.type = "se";
142 		}
143 		
144 		var targetW,targetH,tagetX,targetY;
145 		if(this.target instanceof Alipw.Component){
146 			targetW = this.target.getWidth();
147 			targetH = this.target.getHeight();
148 			targetX = this.target.getX();
149 			targetY = this.target.getY();
150 		}else{
151 			targetW = this.targetEl.width();
152 			targetH = this.targetEl.height();
153 			targetX = this.targetEl.offset().left;
154 			targetY = this.targetEl.offset().top;
155 		}
156 		
157 		var p_topleft = [targetX, targetY];
158 		var p_topright = [targetX + targetW, targetY];
159 		var p_bottomleft = [targetX, targetY + targetH];
160 		var p_bottomright = [targetX + targetW, targetY + targetH];
161 		
162 		this.originalPoints = [p_topleft, p_topright, p_bottomleft, p_bottomright];
163 		
164 		if(this.target instanceof Alipw.Component && this.target.position == "fixed"){
165 			this.anchor = [e.clientX, e.clientY];
166 		}else{
167 			this.anchor = [e.pageX, e.pageY];
168 		}
169 	},
170 	//private
171 	move_ResizeSupporter:function(e){
172 		if(!this.enabled || !this.ready)return;
173 		e.preventDefault();
174 
175 		if(!this.resizing){
176 			this.resizing = true;
177 			
178 			if(!this.resizeProxy){
179 				this.resizeProxy = new Alipw.BorderContainer({
180 					cls:"alipw-resizeproxy",
181 					width:this.targetEl.width(),
182 					height:this.targetEl.height(),
183 					floating:true,
184 					position:(this.target instanceof Alipw.Component)?this.target.position:""
185 				});
186 				
187 				this.addEventListener('destroy',function(){
188 					if(this.resizeProxy){
189 						this.resizeProxy.destroy();
190 						this.resizeProxy = null;
191 					}
192 				},this);
193 			}
194 			
195 			this.resizeProxy.el.css("cursor",this.type + "-resize");
196 			Alipw.ComponentManager.bringToFront(this.resizeProxy);
197 			this.resizeProxy.show();
198 			
199 			this.resizeProxy.setPosition(this.originalPoints[0][0],this.originalPoints[0][1]);
200 		}
201 		
202 		var mouseXY = this.resizeProxy.position == "fixed"?[e.clientX,e.clientY]:[e.pageX,e.pageY];
203 		
204 		var tWidth,tHeight,baseSpot,needRelocatingX,needRelocatingY;
205 		if(this.type == "se"){
206 			tWidth = mouseXY[0] - this.originalPoints[0][0] + (this.originalPoints[3][0] - this.anchor[0]);
207 			tHeight = mouseXY[1] - this.originalPoints[0][1] + (this.originalPoints[3][1] - this.anchor[1]);
208 		}else if(this.type == "s"){
209 			tHeight = mouseXY[1] - this.originalPoints[0][1] + (this.originalPoints[3][1] - this.anchor[1]);
210 		}else if(this.type == "e"){
211 			tWidth = mouseXY[0] - this.originalPoints[0][0] + (this.originalPoints[3][0] - this.anchor[0]);
212 		}else if(this.type == "nw"){
213 			tWidth = this.originalPoints[3][0] - mouseXY[0] + (this.anchor[0] - this.originalPoints[0][0]);
214 			tHeight = this.originalPoints[3][1] - mouseXY[1] + (this.anchor[1] - this.originalPoints[0][1]);
215 			needRelocatingX = true;
216 			needRelocatingY = true;
217 			baseSpot = this.originalPoints[3];
218 		}else if(this.type == "n"){
219 			tHeight = this.originalPoints[3][1] - mouseXY[1] + (this.anchor[1] - this.originalPoints[0][1]);
220 			needRelocatingY = true;
221 			baseSpot = this.originalPoints[2];
222 		}else if(this.type == "w"){
223 			tWidth = this.originalPoints[3][0] - mouseXY[0] + (this.anchor[0] - this.originalPoints[0][0]);
224 			needRelocatingX = true;
225 			baseSpot = this.originalPoints[1];
226 		}else if(this.type == "sw"){
227 			tWidth = this.originalPoints[1][0] - mouseXY[0] + (this.anchor[0] - this.originalPoints[2][0]);
228 			tHeight = mouseXY[1] - this.originalPoints[1][1] + (this.originalPoints[2][1] - this.anchor[1]);
229 			needRelocatingX = true;
230 			baseSpot = this.originalPoints[1];
231 		}else if(this.type == "ne"){
232 			tWidth = mouseXY[0] - this.originalPoints[2][0] + (this.originalPoints[1][0] - this.anchor[0]);
233 			tHeight = this.originalPoints[2][1] - mouseXY[1] + (this.anchor[1] - this.originalPoints[1][1]);
234 			needRelocatingY = true;
235 			baseSpot = this.originalPoints[2];
236 		}
237 		
238 		if(this.minWidth && tWidth < this.minWidth){
239 			tWidth = this.minWidth;
240 		}else if(this.maxWidth && tWidth > this.maxWidth){
241 			tWidth = this.maxWidth;
242 		}
243 		
244 		if(this.minHeight && tHeight < this.minHeight){
245 			tHeight = this.minHeight;
246 		}else if(this.maxHeight && tHeight > this.maxHeight){
247 			tHeight = this.maxHeight;
248 		}
249 		
250 		var newPosX = needRelocatingX?this.resizeProxy.setPosition(baseSpot[0] - tWidth):null;
251 		var newPosY = needRelocatingY?this.resizeProxy.setPosition(null, baseSpot[1] - tHeight):null;
252 		
253 		if(Alipw.isSet(tWidth)){
254 			this.resizeProxy.setWidth(tWidth);
255 		}
256 		if(Alipw.isSet(tHeight)){
257 			this.resizeProxy.setHeight(tHeight);
258 		}
259 		
260 		if(Alipw.isSet(newPosX) || Alipw.isSet(newPosY)){
261 			this.resizeProxy.setPosition(newPosX,newPosY);
262 		}
263 	},
264 	//private
265 	release_ResizeSupporter:function(e){
266 		if(!this.enabled || !this.ready)return;
267 		
268 		if(e.currentTarget.releaseCapture){
269 			e.currentTarget.releaseCapture();  
270 		}else if(window.releaseEvents){  
271 			window.releaseEvents(Event.MOUSEMOVE|Event.MOUSEUP);
272 		}
273 		
274 		if(this.resizing){
275 			this.resizing = false;
276 	
277 			if(this.target instanceof Alipw.Component){
278 				this.target.setWidth(this.resizeProxy.getWidth());
279 				this.target.setHeight(this.resizeProxy.getHeight());
280 				if(this.target.el.css('position') == 'fixed' || this.target.el.css('position') == 'absolute'){
281 					this.target.setPosition(this.resizeProxy.getX(),this.resizeProxy.getY());
282 				}
283 			}else{
284 				this.targetEl.width(this.resizeProxy.getWidth() - (this.targetEl.outerWidth() - this.targetEl.width()));
285 				this.targetEl.height(this.resizeProxy.getHeight() - (this.targetEl.outerHeight() - this.targetEl.height()));
286 				if(this.targetEl.css('position') == 'fixed' || this.targetEl.css('position') == 'absolute'){
287 					this.targetEl.offset({
288 						x:this.resizeProxy.getX(),
289 						y:this.resizeProxy.getY()
290 					});
291 				}
292 			}
293 		}
294 		if(this.resizeProxy){
295 			this.resizeProxy.setVisible(false);
296 		}
297 		this.ready = false;
298 	},
299 	//private
300 	getHandledPosition_ResizeSupporter:function(left,top){
301 		var jWin = jQuery(window);
302 		var winWidth = jWin.width();
303 		var winHeight = jWin.height();
304 		var docWidth = Alipw.getDocWidth();
305 		var docHeight = Alipw.getDocHeight();
306 		var targetWidth,targetHeight; 
307 		targetWidth = this.targetEl.outerWidth();
308 		targetHeight = this.targetEl.outerHeight();
309 
310 		
311 		if(this.target.position == "fixed"){
312 			if(left < 10){
313 				left = 10;
314 			}else if(left + targetWidth > winWidth - 10){
315 				left = winWidth - targetWidth - 10;
316 			}
317 			
318 			if(top < 10){
319 				top = 10;
320 			}else if(top + targetHeight > winHeight - 10){
321 				top = winHeight - targetHeight - 10;
322 			}
323 		}else{
324 			var conWidth = winWidth > docWidth? winWidth : docWidth;
325 			var conHeight = winHeight > docHeight? winHeight : docHeight;
326 			
327 			if(left < 10){
328 				left = 10;
329 			}else if(left + targetWidth > conWidth - 10){
330 				left = conWidth - targetWidth - 10;
331 			}
332 
333 			if(top < 10){
334 				top = 10;
335 			}else if(top + targetHeight > conHeight - 10){
336 				top = conHeight - targetHeight - 10;
337 			}
338 		}
339 		
340 		return [left,top];
341 	}
