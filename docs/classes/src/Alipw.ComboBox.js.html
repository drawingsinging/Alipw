139 	 * @property
140 	 * @type String

144 	name:'',
145 	/**
146 	 * @property
147 	 * @type Mixed HTML Element/Selector
148 	 * @default null

151 	applyTo:null,
152 	//read only
153 	fieldEl:null,
154 	constructor:function(){
155 		Alipw.ComboBox.superclass.constructor.apply(this,arguments);
156 	},
157 	commitProperties:function(){
158 		Alipw.ComboBox.superclass.commitProperties.apply(this,arguments);
159 		if(this.applyTo){
160 			this.fieldEl = Alipw.convertEl(this.applyTo);
161 			if(this.fieldEl[0]){
162 				var data = new Array();
163 				this.displayField = 'text';
164 				this.valueField = 'value';
165 				this.fieldEl.find('option').each(function(i,el){
166 					
167 					var value = $(el).attr('value');
168 					
169 					if((Alipw.isIE6 || Alipw.isIE7) && el.attributes['value'].specified == false){
170 						value = el.innerHTML;
171 					}else if(!Alipw.isSet(value)){
172 						value = el.innerHTML;
173 					}
174 					
175 					data.push({
176 						text:el.innerHTML,
177 						value:value
178 					});
179 				});
180 				this.store = new Alipw.DataStore({
181 					'data':data
182 				});
183 				this.renderTo = this.fieldEl.parent();
184 				this.value = this.fieldEl.val();
185 				
186 				//IE6 BUG. can't set js rendered options selected when the select is hidden 
187 				if(Alipw.isIE6){
188 					this.fieldEl.css({
189 						position:'absolute',
190 						top:'-99999999px',
191 						left:'-99999999px'
192 					});
193 				}else{
194 					this.fieldEl.hide();
195 				}
196 			}
197 		}
198 	},
199 	initialize:function(){
200 		Alipw.ComboBox.superclass.initialize.apply(this,arguments);
201 	},
202 	createDom:function(){
203 		Alipw.ComboBox.superclass.createDom.apply(this,arguments);
204 		this.appendChild(new Alipw.Template(
205 			['<div class="{$baseCls}-text"></div>']
206 		).set({'baseCls':this.baseCls}).compile());
207 	},
208 	renderComplete:function(){
209 		Alipw.ComboBox.superclass.renderComplete.apply(this,arguments);	
210 		
211 		this.addEventListener('click',this.clickHandler_ComboBox,this,true);
212 		
213 		var documentClickHandler = Alipw.createFuncProxy(this.documentClickHandler_ComboBox,this);
214 		var windowResizeHandler = Alipw.createFuncProxy(this.windowResizeHandler_ComboBox,this);
215 		var documentMouseWheelScrollHandler = Alipw.createFuncProxy(this.documentMouseWheelScrollHandler_ComboBox,this);
216 		
217 		Alipw.getDoc().bind('click',documentClickHandler);
218 		Alipw.getWin().bind('resize',windowResizeHandler);
219 		if(Alipw.isIE || Alipw.isOpera || Alipw.isWebKit){
220 			Alipw.getDoc().bind('mousewheel',documentMouseWheelScrollHandler);
221 		}else{
222 			Alipw.getDoc().bind('DOMMouseScroll',documentMouseWheelScrollHandler);
223 		}
224 		this.addEventListener('destroy',function(){
225 			Alipw.getDoc().unbind('click',documentClickHandler);
226 			Alipw.getWin().unbind('resize',windowResizeHandler);
227 			if(Alipw.isIE || Alipw.isOpera || Alipw.isWebKit){
228 				Alipw.getDoc().unbind('mousewheel',documentMouseWheelScrollHandler);
229 			}else{
230 				Alipw.getDoc().unbind('DOMMouseScroll',documentMouseWheelScrollHandler);
231 			}
232 			this.fieldEl = null;
233 		},this);
234 		
235 		if(this.fieldEl && this.fieldEl[0]){
236 			this.el.insertBefore(this.fieldEl);
237 		}else{
238 			this.fieldEl = jQuery('<input type="hidden" name="' + this.name + '" style="display:none" value="' + (Alipw.isSet(this.value)?this.value:'') + '" />');
239 			this.fieldEl.insertAfter(this.el);
240 		}
241 		
242 		this.setValue(this.value);
243 	},
244 	//protected
245 	_doLayout:function(){
246 		Alipw.ComboBox.superclass._doLayout.apply(this,arguments);
247 	},
248 	/**
249 	 * @public
250 	 * @description 展开下拉菜单
251 	 */
252 	expand:function(){
253 		if(this.expanded)return;
254 		
255 		this.__isExpanding = true;
256 		if(!this.list){
257 			this.list = new Alipw.List({
258 				floating:true,
259 				createFieldEl:false,
260 				showShadow:this.showListShadow,
261 				subCls:'alipw-combobox-list',
262 				cls:this.listCls,
263 				maxHeight:this.listMaxHeight,
264 				store:this.store,
265 				width:this.width,
266 				displayField:this.displayField,
267 				valueField:this.valueField,
268 				value:this.value
269 			});
270 			
271 			this.list.addEventListener('itemclick',this.listItemClickHandler_ComboBox,this);
272 			this.list.addEventListener('select',this.listSelectHandler_ComboBox,this);
273 			this.list.addEventListener('change',this.listChangeHandler_ComboBox,this);
274 			this.list.addEventListener('valueChange',this.listValueChangeHandler_ComboBox,this);
275 			this.addEventListener('destroy',function(){
276 				if(this.list){
277 					this.list.destroy();
278 				}
279 			},this);
280 		}
281 		
282 		this.list.setVisible(true);
283 		this.list.setWidth(this.getWidth());
284 		
285 		var selectedListItem = this.list.selectedItems[0];
286 		if(selectedListItem){
287 			this.list.getBody().scrollTop(this.list.selectedItems[0].el[0].offsetTop - parseInt((this.list.getHeight() - selectedListItem.getHeight())/2));
288 		}
289 		
290 		var comboX = this.getX();
291 		var comboY = this.getY();
292 		var x = comboX;
293 		var y = comboY + this.getHeight();
294 		var winHeight = jQuery(window).height();
295 		var listHeight = this.list.getHeight();
296 		
297 		if(winHeight - (y - jQuery(document).scrollTop()) < listHeight + 10){
298 			this.list.el.addClass(this.listAboveCls);
299 			this.el.addClass(this.expandedUpCls);
300 			this.list.el.removeClass(this.listUnderneathCls);
301 			this.el.removeClass(this.expandedDownCls);
302 			
303 			y = comboY - listHeight;
304 		}else{
305 			this.list.el.removeClass(this.listAboveCls);
306 			this.el.removeClass(this.expandedUpCls);
307 			this.list.el.addClass(this.listUnderneathCls);
308 			this.el.addClass(this.expandedDownCls);
309 		}
310 
311 		this.list.setPosition(Math.round(x),Math.round(y));
312 
313 		Alipw.ComponentManager.bringToFront(this.list);
314 		
315 		this.expanded = true;
316 		this.el.addClass(this.expandedCls);
317 		this.__isExpanding = false;
318 	},
319 	/**
320 	 * @public
321 	 * @description 收起下拉菜单
322 	 */
323 	collapse:function(){
324 		if(!this.expanded)return;
325 		
326 		if(this.list){
327 			this.list.setVisible(false);
328 		}
329 		this.expanded = false;
330 		this.el.removeClass(this.expandedCls);
331 		this.el.removeClass(this.expandedUpCls);
332 		this.el.removeClass(this.expandedDownCls);
333 	},
334 	/**
335 	 * @public
336 	 * @description 设置combo box中的显示文本
337 	 * @param {String} text 文本内容
338 	 */
339 	setText:function(text){
340 		this.el.find('.' + this.baseCls + '-text').text(text);
341 	},
342 	/**
343 	 * @public
344 	 * @description 设置combo box中的值
345 	 * @param {String} value 值
346 	 */
347 	setValue:function(value){
348 		var valueChanged = value != this.value;
349 		this.applyValue_ComboBox(value);
350 		
351 		if(this.list){
352 			this.list.setValue(value);
353 		}else if(valueChanged){
354 			this.fireEvent('valueChange',{},false);
355 			this.fieldEl.triggerHandler('alipw-valueChange');
356 		}
357 	},
358 	/**
359 	 * @public
360 	 * @description 设置combo box中的数据寄存器
361 	 * @param {Alipw.DataStore} store 数据寄存器
362 	 */
363 	setStore:function(store){
364 		this.store = store;
365 		this.updateStore();
366 	},
367 	/**
368 	 * @public
369 	 * @description 更新数据寄存器
370 	 */
371 	updateStore:function(){
372 		if(!this.store)return;
373 		var data = this.store.getData();
374 		
375 		var value = data[0] ? data[0][this.valueField]:'';
376 		
377 		if(this.fieldEl[0].nodeName == 'SELECT'){
378 			this.fieldEl.empty();
379 			Alipw.each(data,function(i,item){
380 				this.fieldEl.append('<option value="' + item[this.valueField] + '">' + item[this.displayField] + '</option>');
381 			},this);
382 		}
383 		
384 		this.setValue(value);
385 		if(this.list){
386 			this.collapse();
387 			this.list.destroy();
388 			this.list = null;
389 			//this.list.setStore(this.store);
390 			//this.list.setValue(value);
391 		}
392 	},
393 	//private
394 	applyValue_ComboBox:function(value){
395 		if(!Alipw.isSet(value) || !(this.store instanceof Alipw.DataStore))return;
396 		this.value = null;
397 		this.setText('');
398 		this.fieldEl.val('');
399 		
400 		var data = this.store.getData();
401 		Alipw.each(data,function(i,o){
402 			if(value == o[this.valueField]){
403 				this.setText(o[this.displayField]);
404 				this.value = value;
405 				this.fieldEl.val(value);
406 				return false;//break the loop
407 			}
408 		},this);
409 	},
410 	//private
411 	clickHandler_ComboBox:function(e){
412 		if(this.expanded){
413 			this.collapse();
414 		}else{
415 			this.expand();
416 		}
417 	},
418 	//private
419 	documentClickHandler_ComboBox:function(e){
420 		if(Alipw.isInNode(e.target,this.el[0]) || (this.list && Alipw.isInNode(e.target,this.list.el[0]) )){
421 			return;
422 		}
423 		this.collapse();
424 	},
425 	//private
426 	windowResizeHandler_ComboBox:function(e){
427 		if(this.__isExpanding)return;
428 		this.collapse();
429 	},
430 	//private
431 	documentMouseWheelScrollHandler_ComboBox:function(e){
432 		if(this.list){
433 			if(Alipw.isInNode(e.target,this.list.el[0])){
434 				e.preventDefault();
435 				
436 				var detail;
437 				if(Alipw.isIE || Alipw.isOpera || Alipw.isWebKit){
438 					detail = -1 * e.wheelDelta / 40;
439 				}else{
440 					detail = e.detail;
441 				}
442 				this.list.scrollY(detail * 10);
443 			}else{
444 				this.collapse();
445 			}
446 		}
447 	},
448 	//private
449 	listItemClickHandler_ComboBox:function(e){
450 		this.collapse();
451 	},
452 	//private
453 	listSelectHandler_ComboBox:function(e){
454 		this.fireEvent('select',{
455 			index:e.index,
456 			item:e.item
457 		},false);
458 	},
459 	//private
460 	listValueChangeHandler_ComboBox:function(e){
461 		var list = e.currentTarget;
462 		this.applyValue_ComboBox(list.value);
463 		this.fireEvent('valueChange',{},false);
464 		this.fieldEl.triggerHandler('alipw-valueChange');
465 	},
466 	//private
467 	listChangeHandler_ComboBox:function(e){
468 		this.fireEvent('change',{},false);
469 		this.fieldEl.triggerHandler('change');
470 	}
