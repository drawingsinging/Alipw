1 /**
  2  * @constructor Alipw.Template
  3  * @extends Alipw.Nonvisual
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 模板。用于创建HTML模板，并方便地进行数据替换。
  6  */
  7 
  8 Alipw.Template = Alipw.extend(Alipw.Nonvisual,
  9 /** @lends Alipw.Template.prototype */
 10 {
 11 	constructor:function(){
 12 		Alipw.Template.superclass.constructor.apply(this,arguments);
 13 	},
 14 	commitProperties:function(){
 15 		Alipw.Template.superclass.commitProperties.apply(this,arguments);
 16 		if(typeof(this.config) == "string"){
 17 			this.html = this.config;
 18 		}else if(this.config instanceof Array){
 19 			this.html = this.config.join("");
 20 		}
 21 	},
 22 	initialize:function(){
 23 		Alipw.Template.superclass.initialize.apply(this,arguments);
 24 	},
 25 	set:function(o){
 26 		for(var i in o){
 27 			var exp = new RegExp("\\{\\$" + i + "\\}","g");
 28 			this.html = this.html.replace(exp,o[i]);
 29 		}
 30 		
 31 		return this;
 32 	},
 33 	compile:function(){
 34 		this.el = jQuery(this.html);
 35 		return this.el;
 36 	}
 37 });
