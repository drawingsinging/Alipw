 10 Alipw.Slider = Alipw.extend(Alipw.Component,
 11 /** @lends Alipw.Slider.prototype */
 12 {
 13 	/**
 14 	 * @property
 15 	 * @type Number
 16 	 * @default 200
 17 	 * @description [config option]定义Slider的宽度。
 18 	 */
 19 	width:200,
 20 	/**
 21 	 * @property
 22 	 * @type String
 23 	 * @default 'horizontal'
 24 	 * @description [config option]定义Slider的类型（水平或竖直），可选值有'horizontal','vertical'。
 25 	 */
 26 	type:"horizontal",
 27 	/**
 28 	 * @property
 29 	 * @type String
 30 	 * @default 'head'
 31 	 * @description [config option]定义Slider值域的方向。可选值有'head','foot'。若为'head'，则水平模式的左侧为最小值，右侧为最大值，垂直方向的顶部为最小值，底部为最大值；若为'foot'，则反之。
 32 	 */
 33 	valueFrom:"head",
 34 	/**
 35 	 * @property
 36 	 * @type Number
 37 	 * @default 0
 38 	 * @description [config option][property]作为配置选项，定义Slider的初始值。作为属性，指示Slider当前的值。
 39 	 */
 40 	value:0,
 41 	/**
 42 	 * @property
 43 	 * @type Number
 44 	 * @default 0
 45 	 * @description [config option]作为配置选项，定义Slider的初始值。作为属性，指示Slider当前的值。
 46 	 */
 47 	values:null,
 48 	/**
 49 	 * @property
 50 	 * @type Number
 51 	 * @default 0
 52 	 * @description [config option]定义Slider值域内的最小步长。若为0，则视为不设置最小步长。此设置仅在values属性为设置或者values数组长度为0时才有效。
 53 	 */
 54 	step:0,
 55 	/**
 56 	 * @property
 57 	 * @type Boolean
 58 	 * @default true
 59 	 * @description [config option]定义是否填充选值区域。
 60 	 */
 61 	showFill:true,
 62 	/**
 63 	 * @property
 64 	 * @type Number
 65 	 * @default 0
 66 	 * @description [config option]定义Slider的起始值。
 67 	 */
 68 	minValue:0,
 69 	/**
 70 	 * @property
 71 	 * @type Number
 72 	 * @default 100
 73 	 * @description [config option]定义Slider的终点值。
 74 	 */
 75 	maxValue:100,
 76 	/**
 77 	 * @property
 78 	 * @type String
 79 	 * @default 'alipw-slider'
 80 	 * @description [config option]定义class名称的基本前缀。
 81 	 */
 82 	baseCls:"alipw-slider",
 83 	constructor:function(){
 84 		Alipw.Slider.superclass.constructor.apply(this,arguments);
 85 	},
 86 	commitProperties:function(){
 87 		Alipw.Slider.superclass.commitProperties.apply(this,arguments);
 88 		if(!this.values){
 89 			this.values = new Array();
 90 		}
 91 		
 92 		if(this.values.length == 0 && this.step){
 93 			var len = parseInt((this.maxValue - this.minValue) / this.step);
 94 			this.values = new Array();
 95 			for(var i=1;i<=len;i++){
 96 				this.values.push(this.minValue + i * this.step);
 97 			}
 98 		}
 99 	},
100 	initialize:function(){
101 		Alipw.Slider.superclass.initialize.apply(this,arguments);
102 		this.sliderDragManager = new Object();//name space
103 	},
104 	createDom:function(){
105 		Alipw.Slider.superclass.createDom.apply(this,arguments);
106 		this.el.append(['<div class="' + this.baseCls + '-wrap">',
107 			                '<div class="' + this.baseCls + '-track">',
108 	                     		'<span class="' + this.baseCls + '-track-left">',
109 	                     			'<span class="' + this.baseCls + '-track-right">',
110 	                     				'<span class="' + this.baseCls + '-track-center">',
111 	                     				'</span>',
112 	                     			'</span>',
113 	                     		'</span>',
114 	                     		'<div class="' + this.baseCls + '-track-fill">',
115 		                     		'<span class="' + this.baseCls + '-track-fill-left">',
116 		                     			'<span class="' + this.baseCls + '-track-fill-right">',
117 		                     				'<span class="' + this.baseCls + '-track-fill-center">',
118 		                     				'</span>',
119 		                     			'</span>',
120 		                     		'</span>',
121 	                     		'</div>',
122 	                     	'</div>',
123 	                     	'<div class="' + this.baseCls + '-block">',
124 		                     	'<span class="' + this.baseCls + '-block-left">',
125 	                     			'<span class="' + this.baseCls + '-block-right">',
126 	                     				'<span class="' + this.baseCls + '-block-center">',
127 		                     				'<span class="' + this.baseCls + '-block-bg">',
128 		                     				'</span>',
129 	                     				'</span>',
130 	                     			'</span>',
131 	                     		'</span>',
132 	                     	'</div>',
133                      	'</div>'].join(""));
134 		
135 		if(this.type == "vertical"){
136 			this.el.find("." + this.baseCls + "-wrap").addClass("alipw-slider-vertical");
137 		}
138 		
139 		this._block = this.el.find("." + this.baseCls + "-block");
140 		this._fill = this.el.find("." + this.baseCls + "-track-fill");
141 		this._track = this.el.find("." + this.baseCls + "-track");
142 		
143 		var _this = this;
144 		
145 		this._track.setHeight = function(height){
146 			if(_this.type !== "vertical"){
147 				return;
148 			}
149 			
150 			var innerCon = _this.el.find("." + _this.baseCls + "-track-center");
151 			var innerHeight =  innerCon.height();
152 			var outerHeight = _this.el.find("." + _this.baseCls + "-track").outerHeight();
153 			var d = outerHeight - innerHeight;
154 			innerCon.height(height - d);
155 		};
156 		
157 		this._fill.setHeight = function(height){
158 			if(_this.type !== "vertical"){
159 				return;
160 			}
161 			
162 			var innerCon = _this.el.find("." + _this.baseCls + "-track-fill-center");
163 			var innerHeight =  innerCon.height();
164 			var outerHeight = _this.el.find("." + _this.baseCls + "-track-fill").outerHeight();
165 			var d = outerHeight - innerHeight;
166 			innerCon.height(height - d);
167 		};
168 		
169 		this._block.setHeight = function(height){
170 			if(_this.type !== "vertical"){
171 				return;
172 			}
173 			
174 			var innerCon = _this.el.find("." + _this.baseCls + "-block-center");
175 			var innerHeight =  innerCon.height();
176 			var outerHeight = _this.el.find("." + _this.baseCls + "-block").outerHeight();
177 			var d = outerHeight - innerHeight;
178 			innerCon.height(height - d);
179 		};
180 		
181 		this._block.centerBg = function(){
182 			var blockInnerHeight = _this.el.find("." + _this.baseCls + "-block-center").height();
183 			var bg = _this.el.find("." + _this.baseCls + "-block-bg");
184 			bg.css("margin-top",parseInt((blockInnerHeight - bg.height())/2));
185 		};
186 	},
187 	renderComplete:function(){
188 		Alipw.Slider.superclass.renderComplete.apply(this,arguments);
189 		
190 		var _this = this;
191 		if(!this.showFill){
192 			this._fill.hide();
193 		}
194 		
195 		if(this.valueFrom == "foot"){
196 			if(this.type == "vertical"){
197 				this._fill.css({
198 					top:"auto",
199 					bottom:"0"
200 				});
201 			}else{
202 				this._fill.css({
203 					left:"auto",
204 					right:"0"
205 				});
206 			}
207 		}
208 		
209 		this._block.hover(function(e){
210 			$(e.currentTarget).addClass(_this.baseCls + "-block-hover");
211 		},function(e){
212 			$(e.currentTarget).removeClass(_this.baseCls + "-block-hover");
213 		});
214 		
215 		this._track.bind("mousedown",jQuery.proxy(this.trackMouseDownHander_Slider,this));
216 		
217 		this._block.bind("mousedown",function(e){
218 			e.preventDefault();
219 			$(e.currentTarget).addClass(_this.baseCls + "-block-down");
220 			
221 			_this.getReady_Slider(e);
222 		});
223 		
224 		jQuery(document).bind("mousemove",jQuery.proxy(this.move_Slider,this));
225 		jQuery(document).bind("mouseup",jQuery.proxy(this.release_Slider,this));
226 		
227 		jQuery(document).bind("mouseup",function(e){
228 			_this._block.removeClass(_this.baseCls + "-block-down");
229 		});
230 		
231 		this.setValue(this.value);
232 	},
233 	//protected
234 	_doLayout:function(){
235 		Alipw.Slider.superclass._doLayout.apply(this,arguments);
236 		if(this.width){
237 			if(this.type == "vertical"){
238 				this.el.height(this.width);
239 			}else{
240 				this.el.width(this.width);
241 			}
242 		}
243 		
244 		if(this.type == "vertical"){
245 			//set the height of the track
246 			this._track.setHeight(this.width);
247 			
248 			//center the block background
249 			this._block.centerBg();
250 		}
251 	},
252 	setValue:function(value){
253 		this.applyValue_Slider(value);
254 	},
255 	//private
256 	applyValue_Slider:function(value,isUserOperation){
257 		if(value > this.maxValue){
258 			value = this.maxValue;
259 		}else if(value < this.minValue){
260 			value = this.minValue;
261 		}
262 		
263 		this.adjustBlockPosition_Slider(value);
264 		
265 		if(value != this.value){
266 			this.value = value;
267 			this.fireEvent("valueChange",{
268 				value:value
269 			},false);
270 			
271 			if(isUserOperation){
272 				this.fireEvent("change",{
273 					value:value
274 				},false);
275 			}
276 		}
277 	},
278 	//private
279 	adjustBlockPosition_Slider:function(value){
280 		var left = this.getCoordXByValue_Slider(value);
281 
282 		if(this.type == "vertical"){
283 			this._block.offset({
284 				top:left
285 			});
286 			this._block.css('top',parseInt(this._block.css('top')) + 'px');
287 			if(this.valueFrom == "foot"){
288 				this._fill.setHeight(this.el.offset().top + this.el.height() - left - parseInt(this._block.outerHeight()/2));
289 			}else{
290 				this._fill.setHeight(left - this.el.offset().top + parseInt(this._block.outerHeight()/2));
291 			}
292 		}else{
293 			this._block.offset({
294 				left:left
295 			});
296 			this._block.css('left',parseInt(this._block.css('left')) + 'px');
297 			
298 			if(this.valueFrom == "foot"){
299 				this._fill.width(this.el.offset().left + this.el.width() - left - parseInt(this._block.outerWidth()/2));
300 			}else{
301 				this._fill.width(left - this.el.offset().left + parseInt(this._block.outerWidth()/2));
302 			}
303 		}
304 	},
305 	//private
306 	getReady_Slider:function(e){
307 		if(!this.enabled)return;
308 		
309 		e.preventDefault();
310 		this.sliderDragManager.ready = true;
311 		this.sliderDragManager._dx = this._block.offset().left - e.pageX;
312 		this.sliderDragManager._dy = this._block.offset().top - e.pageY;
313 	},
314 	//private
315 	move_Slider:function(e){
316 		if(!this.enabled || !this.sliderDragManager.ready)return;
317 		e.preventDefault();
318 		
319 		var v = this.type == "vertical";
320 
321 		if(!this.sliderDragManager.dragging){
322 			this.sliderDragManager.dragging = true;
323 		}
324 
325 		var value = this.getValueByCoord_Slider(v?(e.pageY + this.sliderDragManager._dy):(e.pageX + this.sliderDragManager._dx));
326 		this.applyValue_Slider(value,true);
327 	},
328 	//private
329 	release_Slider:function(e){
330 		if(!this.enabled)return;
331 		
332 		this.sliderDragManager.dragging = false;
333 		this.sliderDragManager.ready = false;
334 	},
335 	//private
336 	getCoordXByValue_Slider:function(value){
337 		var v = this.type == "vertical";
338 		
339 		var blockWidth = v?this._block.outerHeight():this._block.outerWidth();
340 		var slideCoord = v?this.el.offset().top:this.el.offset().left;
341 		var slideWidth = v?this.el.height():this.el.width();
342 		var left;
343 		
344 		if(this.valueFrom == "head"){
345 			left = slideCoord + ((value - this.minValue) / (this.maxValue - this.minValue)) * (slideWidth - blockWidth);
346 		}else if(this.valueFrom == "foot"){
347 			left = slideCoord + slideWidth - blockWidth - ((value - this.minValue) / (this.maxValue - this.minValue)) * (slideWidth - blockWidth);
348 		}
349 		left = parseInt(left);
350 		
351 		return left;
352 	},
353 	//private
354 	getValueByCoord_Slider:function(left){
355 		var v = this.type == "vertical";
356 		
357 		var blockWidth = v?this._block.outerHeight():this._block.outerWidth();
358 		var slideCoord = v?this.el.offset().top:this.el.offset().left;
359 		var slideWidth = v?this.el.height():this.el.width();
360 		
361 		if(left < slideCoord){
362 			left = slideCoord;
363 		}else if(left + blockWidth > slideCoord + slideWidth){
364 			left = slideCoord + slideWidth - blockWidth;
365 		}
366 		
367 		var value;
368 		if(this.valueFrom == "head"){
369 			value = this.minValue + (this.maxValue - this.minValue) * ((left - slideCoord)/(slideWidth - blockWidth));
370 		}else if(this.valueFrom == "foot"){
371 			value = this.minValue + (this.maxValue - this.minValue) * ((slideCoord + slideWidth - blockWidth - left)/(slideWidth - blockWidth));
372 		}
373 		
374 		var d = -1;//for chrome weird bug.
375 		var nearestValue;
376 		if(this.values.length > 0){
377 			for(var i=0,len=this.values.length;i<len;i++){
378 				if(this.values[i] > this.minValue && this.values[i] < this.maxValue ){
379 					if(d == -1 || Math.abs(value - this.values[i]) < d){		
380 						d = Math.abs(value - this.values[i]);
381 						nearestValue = this.values[i];
382 					}
383 				}
384 			}
385 			
386 			if(Math.abs(value - this.minValue) < d){
387 				d = Math.abs(value - this.minValue);
388 				nearestValue = this.minValue;
389 			}
390 			
391 			if(Math.abs(value - this.maxValue) < d){
392 				d = Math.abs(value - this.maxValue);
393 				nearestValue = this.maxValue;
394 			}
395 			
396 			value = nearestValue;
397 		}
398 		
399 		return value;
400 	},
401 	//private
402 	trackMouseDownHander_Slider:function(e){
403 		var v = this.type == "vertical";
404 		var blockSize = [this._block.outerWidth(),this._block.outerHeight()];
405 		var value = this.getValueByCoord_Slider(v?(e.pageY - parseInt(blockSize[1] / 2)):(e.pageX - parseInt(blockSize[0] / 2)));
406 		this.applyValue_Slider(value,true);
407 	}
