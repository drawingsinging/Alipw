1 /**
  2  * @class
  3  * @type singleton
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 可视控件管理器。负责管理所有被创建的可视控件，对其ID，层级顺序以及生命周期中涉及的资源的创建和释放进行统一管理。
  6  */
  7 Alipw.ComponentManager = function(){
  8 	var cs = new Object(); // component storage
  9 	var fs = new Array(); // floating component storage
 10 	var is = new Array(); // auto ID component storage
 11 
 12 	var pub = 
 13 	/** @lends Alipw.ComponentManager */
 14 	{
 15 		/**
 16 		 * @property
 17 		 * @type Number
 18 		 * @description 浮动Component的z-index的初始值
 19 		 * */
 20 		zseed:8000,
 21 		
 22 		/**
 23 		 * @public
 24 		 * @param {Alipw.Component} component 需要被注册的Alipw.Component实例
 25 		 * @description 将制定component实例注册到管理器
 26 		 * */
 27 		register:function(component){
 28 			if(component.floating && component.floatingManagement){
 29 				fs.push(component);
 30 				component.zIndex = this.zseed + fs.length * 5;
 31 			}
 32 			if(!component.id){
 33 				is.push(component);
 34 				component.id = "Alipw-comp-" + is.length;
 35 			}
 36 			cs[component.id] = component;
 37 		},
 38 		/**
 39 		 * @public
 40 		 * @param {Alipw.Component} component 需要被注销的Alipw.Component实例
 41 		 * @description 将制定component实例从管理器注销
 42 		 * */
 43 		unregister:function(component){
 44 			Alipw.removeItemFromArray(component, fs);
 45 			this.updateZIndex();
 46 			
 47 			//don't change id stack.
 48 			for(var i=0,len=is.length;i<len;i++){
 49 				if(is[i] == component){
 50 					delete is[i];
 51 				}
 52 			}
 53 			
 54 			delete cs[component.id];
 55 		},
 56 		/**
 57 		 * @public
 58 		 * @param {Integer} id component的id
 59 		 * @description 通过id获取已创建的component实例。
 60 		 * */
 61 		getComponent:function(id){
 62 			return cs[id];
 63 		},
 64 		/**
 65 		 * @public
 66 		 * @param {Alipw.Component} component component实例
 67 		 * @description 将指定的component实例的显示层级顺序提到最前，此方法仅对floating属性和floatingManagement属性都为true的component实例有效。
 68 		 * */
 69 		bringToFront:function(component){
 70 			if(component.floating && component.floatingManagement){
 71 				Alipw.removeItemFromArray(component, fs);
 72 				fs.push(component);
 73 				this.updateZIndex();
 74 			}
 75 		},
 76 		sendToBack:function(){
 77 			
 78 		},
 79 		updateZIndex:function(){
 80 			for(var i=0,len=fs.length;i<len;i++){
 81 				fs[i].zIndex = this.zseed + (i + 1) * 5;
 82 				if(fs[i].rendered){
 83 					fs[i].setPosition(null,null,fs[i].zIndex);
 84 				}
 85 			}
 86 		}
 87 	};
 88 	
 89 	return pub;
 90 }();
