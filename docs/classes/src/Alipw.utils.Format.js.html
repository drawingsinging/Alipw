 15 		/**
 16 		* YY,YYYY - year
 17 		* MM,M - month
 18 		* DD,D - day
 19 		*/
 20 		date:function(date,format){
 21 			date = new Date(date);
 22 			if(isNaN(date) || date == 'Invalid Date'){
 23 				return 'Invalid Date';
 24 			}
 25 			
 26 			var year = date.getFullYear().toString();
 27 			var month = (date.getMonth() + 1).toString();
 28 			var day = date.getDate().toString();
 29 			
 30 			var YY = year.substr(2,2);
 31 			var YYYY = year;
 32 			var MM = month.length < 2 ? '0' + month : month;
 33 			var M = month;
 34 			var DD = day.length < 2 ? '0' + day : day;
 35 			var D = day;
 36 			
 37 			var output = format;
 38 			output = output.replace(/YYYY/g,YYYY);
 39 			output = output.replace(/YY/g,YY);
 40 			output = output.replace(/MM/g,MM);
 41 			output = output.replace(/M/g,M);
 42 			output = output.replace(/DD/g,DD);
 43 			output = output.replace(/D/g,D);
 44 			
 45 			return output;
 46 		}
 47 	};
 48 	
 49 	return pub;
 50 }();
