1 /**
  2  * @constructor Alipw.List
  3  * @extends Alipw.BorderContainer
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 数据展示列表。用于展示列表型数据的容器。
  6  * @example
  7  * 
  8  */
  9 
 10 Alipw.List = Alipw.extend(Alipw.BorderContainer,
 11 /** @lends Alipw.List.prototype */
 12 {
 13 	baseCls:'alipw-list',
 14 	selectedCls:'alipw-listitem-selected',
 15 	scroll:true,
 16 	store:null,
 17 	selectable:true,
 18 	multipleSelection:false,
 19 	displayField:'text',
 20 	valueField:'value',
 21 	value:null,
 22 	width:200,
 23 	maxHeight:null,
 24 	createFieldEl:true,
 25 	applyTo:null,
 26 	commitProperties:function(){
 27 		Alipw.List.superclass.commitProperties.apply(this,arguments);
 28 	},
 29 	initialize:function(){
 30 		Alipw.List.superclass.initialize.apply(this,arguments);
 31 		this.listItems = new Array();
 32 		this.selectedItems = new Array();
 33 	},
 34 	createDom:function(){
 35 		Alipw.List.superclass.createDom.apply(this,arguments);
 36 		this.renderListItem();
 37 	},
 38 	renderComplete:function(){
 39 		Alipw.List.superclass.renderComplete.apply(this,arguments);
 40 		
 41 		if(this.scroll){
 42 			this.getBody().css('overflow-y','auto');
 43 		}
 44 	},
 45 	//protected
 46 	_doLayout:function(){
 47 		Alipw.List.superclass._doLayout.apply(this,arguments);
 48 		
 49 		//for IE6,7 bug
 50 		if(Alipw.isIE6 || Alipw.isIE7){
 51 			if(this.getBody()[0].style.height == 'auto'){
 52 				this.getBody().css('overflow-y','hidden');
 53 			}else if(this.scroll){
 54 				this.getBody().css('overflow-y','auto');
 55 			}
 56 		}
 57 	},
 58 	renderListItem:function(){
 59 		if(!this.store instanceof Alipw.DataStore) return;
 60 		
 61 		var data = this.store.getData();
 62 		Alipw.each(data,function(i,o){
 63 			var listItem = new Alipw.ListItem({
 64 				text:data[i][this.displayField],
 65 				value:data[i][this.valueField],
 66 				data:data[i],
 67 				selectedCls:this.selectedCls,
 68 				index:i,
 69 				renderTo:this
 70 			});
 71 			
 72 			if(this.value && this.value == data[i][this.valueField]){
 73 				if(this.multipleSelection || (!this.multipleSelection && this.selectedItems.length == 0)){
 74 					listItem.select();
 75 					this.selectedItems.push(listItem);
 76 				}
 77 			}
 78 			listItem.addEventListener('itemclick',this.itemClickHandler_List,this);
 79 			listItem.addEventListener('select',this.selectHandler_List,this);
 80 			this.listItems.push(listItem);
 81 			
 82 		},this);
 83 	},
 84 	setValue:function(value){
 85 		if(Alipw.isSet(value) && this.listItems.length > 0){
 86 			var muti = this.multipleSelection;
 87 			var valueArray = muti?value.split(','):null;
 88 			
 89 			var foundSelected = false;
 90 			Alipw.each(this.listItems,function(i,item){
 91 				if(!muti){
 92 					if(item.value == value && !foundSelected){
 93 						foundSelected = true;
 94 						item.select();
 95 					}else{
 96 						item.unselect();
 97 						Alipw.removeItemFromArray(item,this.selectedItems);
 98 					}
 99 				}else{
100 					if(Alipw.indexOfArray(item.value,valueArray) != -1){
101 						item.select();
102 					}
103 				}
104 			},this);
105 			
106 			if(!muti){
107 				if(!foundSelected){
108 					this.value = '';
109 				}
110 			}else{
111 				
112 			}
113 		}
114 	},
115 	setStore:function(store){
116 		if(store instanceof Alipw.DataStore){
117 			this.store = store;
118 			this.updateListItems();
119 		}
120 	},
121 	updateListItems:function(){
122 		Alipw.each(this.listItems,function(index,el){
123 			el.destroy();
124 		});
125 		this.listItems.splice(0,999999999);
126 		this.selectedItems.splice(0,99999999);
127 		
128 		this.renderListItem();
129 	},
130 	//private
131 	applyValue_List:function(value,fireChangeEvent){
132 		var oldValue = this.value;
133 		this.value = value;
134 		if(oldValue != value){
135 			this.fireEvent('valueChange',{},false);
136 			if(fireChangeEvent){
137 				this.fireEvent('change',{},false);
138 			}
139 		}
140 	},
141 	//private
142 	itemClickHandler_List:function(e){
143 		var listItem = e.currentTarget;
144 		
145 		var oldValue = this.value;
146 		if(this.selectable){
147 			listItem.select(true);
148 		}
149 		this.fireEvent('itemclick',{
150 			index:listItem.index,
151 			item:listItem
152 		},false);
153 	},
154 	//private
155 	selectHandler_List:function(e){
156 		var listItem = e.currentTarget;
157 		
158 		if(!this.multipleSelection){
159 			if(Alipw.indexOfArray(listItem,this.selectedItems) != -1)return;
160 			Alipw.each(this.selectedItems,function(index,item){
161 				item.unselect();
162 				Alipw.removeItemFromArray(item,this.selectedItems);
163 			},this);
164 		}else{
165 			if(Alipw.indexOfArray(listItem,this.selectedItems) != -1){
166 				listItem.unselect();
167 				Alipw.removeItemFromArray(listItem,this.selectedItems);
168 				return;
169 			}
170 		}
171 
172 		this.selectedItems.push(listItem);
173 		
174 		if(!this.multipleSelection){
175 			this.applyValue_List(listItem.value,e.isUserOperation);
176 		}else{
177 			
178 		}
179 
180 		this.fireEvent('select',{
181 			index:listItem.index,
182 			item:listItem
183 		},false);
184 	}
185 });
