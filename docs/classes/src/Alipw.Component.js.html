 77 		/**
 78 		 * @property
 79 		 * @type String

 83 		downCls:'',
 84 		/**
 85 		 * @property
 86 		 * @type Boolean

146 		inlineStyle:null,

155 			Alipw.apply(this, this.config);
156 		},
157 		initialize:function(){
158 			Alipw.ComponentManager.register(this);
159 		},
160 		createDom:function(){
161 			/**
162 			 * @memberOf Alipw.Component.prototype
163 			 * @property
164 			 * @type JQuery Object
165 			 * @description 一个jQuery对象。其引用的HTML元素为当前component的HTML元素。
166 			 */
167 			this.el = $(document.createElement(this.defaultEl));
168 			
169 			if(this.listeners){
170 				for(var i in this.listeners){
171 					if(this.listeners[i] instanceof Function){
172 						this.addEventListener(i,this.listeners[i],this);
173 					}
174 				}
175 			}
176 			
177 			this.el.attr("id", this.id);
178 			if(this.baseCls){
179 				this.el.attr("class", this.baseCls);
180 			}
181 			if(this.subCls){
182 				this.el.addClass(this.subCls);
183 			}
184 			if(this.cls){
185 				this.el.addClass(this.cls);
186 			}
187 			if(this.zIndex){
188 				this.el.css("z-index",this.zIndex);
189 			}
190 			if(this.position == "fixed"){
191 				if(parseInt(Alipw.ie()) == 6){
192 					this.el.css("position","absolute");
193 					this.x = this.getX() - jQuery(window).scrollLeft();
194 					this.y = this.getY() - jQuery(window).scrollTop();
195 					jQuery(window).bind("scroll",jQuery.proxy(this.scrollHandler_Component,this));
196 					this.addEventListener('destroy',function(e){
197 						jQuery(window).unbind('scroll',this.scrollHandler_Component);
198 					},this);
199 				}else{
200 					this.el.css("position","fixed");
201 				}
202 			}
203 			
204 			if(this.inlineStyle){
205 				this.el.css(this.inlineStyle);
206 			}
207 		},
208 		/**
209 		 * @public
210 		 * @description 将该component渲染到指定的HTML节点中。
211 		 * @param {Mix HTML Element/selector} con 该component要渲染到的HTML节点。如con为空，则渲染到renderTo指定的节点中。如renderTo属性为空，则渲染到body节点中。
212 		 */
213 		render:function(con){
214 			if(!con){
215 				if(this.renderTo){
216 					con = this.renderTo;
217 				}else{
218 					con = document.body;
219 				}
220 			}
221 			
222 			this.fireEvent('beforeRender',{},false);
223 			if(con instanceof Alipw.Container){
224 				//the appended component will call doLayout method and that will make the parent container update the layout. so here we don't make it update layout.
225 				con.appendChild(this,false);
226 			}else{
227 				con = Alipw.convertEl(con);
228 				con.append(this.el);
229 			}
230 			this.removed = false;
231 			this.fireEvent('afterRender',{},false);
232 			
233 			if(!this.rendered){
234 				this.renderComplete();
235 				
236 				this.rendered = true;
237 				this.creationCompleteFn_Component();
238 				this.fireEvent("creationComplete",{},false);
239 			}
240 		},
241 		renderComplete:function(isUpdating){
242 			this.addEventListener('afterLayout',this.afterLayoutHandler_Component,this);
243 			
244 			if(this.visibleRender && this.el.is(':hidden')){
245 				var tempWrapper = jQuery('<div style="position:absolute; left:-99999px; top:-99999px;"></div>');
246 				var nextSibling = this.el.next();
247 				var parent = this.el.parent();
248 				jQuery(document.body).append(tempWrapper);
249 				tempWrapper.append(this.el);
250 				
251 				this.doLayout();
252 				
253 				if(nextSibling[0]){
254 					this.el.insertBefore(nextSibling);
255 				}else{
256 					parent.append(this.el);
257 				}
258 				tempWrapper.remove();
259 			}else{
260 				this.doLayout();
261 			}
262 
263 			//common interface to do more action after the dom element is rendered
264 			
265 			if(this.overCls){
266 				this.addEventListener('mouseover',this.mouseOverHandlerForOverCls_Component,this,true);
267 				this.addEventListener('mouseout',this.mouseOutHandlerForOverCls_Component,this,true);
268 			}
269 			
270 			if(this.downCls){
271 				var mouseUpHandler = Alipw.createFuncProxy(this.mouseUpHandlerForDownCls_Component,this);
272 				
273 				this.addEventListener('mousedown',this.mouseDownHandlerForDownCls_Component,this,true);
274 				jQuery(document).bind('mouseup',mouseUpHandler);
275 				
276 				this.addEventListener('destroy',function(e){
277 					jQuery(document).unbind('mouseup',mouseUpHandler);
278 				},this);
279 			}
280 		},
281 		/**
282 		 * @public
283 		 * @description 为该component添加事件监听
284 		 * @param {String} type 事件类型。
285 		 * @param {Function} fn 监听函数。
286 		 * @param {Object} [scope] 作用域。指定监听函数fn中的this指针所指向的对象。默认为undefined。
287 		 * @param {Boolean} [useBrowserEvent] 是否为浏览器事件。浏览器事件是指click, mouseover等浏览器自带事件。默认为false。
288 		 */
289 		addEventListener:function(type, fn, scope, useBrowserEvent){
290 			Alipw.EventManager.addListener(this,type,fn,scope,useBrowserEvent);
291 		},
292 		/**
293 		 * @public
294 		 * @description 为该component移除事件监听
295 		 * @param {String} type 事件类型。
296 		 * @param {Function} fn 监听函数。
297 		 * @param {Boolean} [useBrowserEvent] 是否为浏览器事件。浏览器事件是指click, mouseover等浏览器自带事件。默认为false。
298 		 */
299 		removeEventListener:function(type, fn, useBrowserEvent){
300 			Alipw.EventManager.removeListener(this,type,fn,useBrowserEvent);
301 		},
302 		/**
303 		 * @public
304 		 * @description 触发该component的指定事件
305 		 * @param {String} type 事件类型。
306 		 * @param {Object} [params] 附带参数。
307 		 * @param {Boolean} [bubble] 是否冒泡,默认为true。
308 		 * @param {Boolean} [useBrowserEvent] 是否为浏览器事件。浏览器事件是指click, mouseover等浏览器自带事件。默认为false。
309 		 */
310 		fireEvent:function(type,params,bubble, useBrowserEvent){
311 			return Alipw.EventManager.fireEvent(this,type,params,bubble,useBrowserEvent);
312 		},
313 		regEvents:function(events){
314 			for(var i=0,len=events.length;i<len;i++){
315 				Rainy.EventManager.register(events[i]);
316 			}
317 		},

320 		},

323 				this.render();
324 			}
325 			this.setVisible(true);
326 		},

329 		},

331 		 * @public
332 		 */

335 			
336 			this.fireEvent('beforeRemove',{},false);
337 			if(removeEventListeners){

343 			this.fireEvent('afterRemove',{},false);
344 		},
345 		/**

351

353 			this.remove(true);
354 			if(this.parentContainer)this.parentContainer = null;// release the parentContainer reference to avoid circular reference
355 			Alipw.ComponentManager.unregister(this);
356 			this.el = null;
357 			this.destroyed = true;
358 		},
359 		/**
360 		 * @public
361 		 * @description 设置component是否可见。
362 		 * @param {Boolean} visible 是否可见。
363 		 * @param {Boolean} [includeInLayout] 是否占位。默认与visible相同。
364 		 * @param {Boolean} [silent] 是否无声。如为true,则当可见属性改变时，不触发visibilityChange事件。默认为false.
365 		 */
366 		setVisible:function(visible,includeInLayout,silent){
367 			if(!Alipw.isSet(includeInLayout)){
368 				includeInLayout = visible;
369 			}
370 			
371 			if(visible){
372 				this.el.show();
373 				this.el.css("visibility","visible");
374 			}else{
375 				this.el.hide();
376 				if(includeInLayout){
377 					this.el.show();
378 					this.el.css({
379 						visibility:"hidden"
380 					});
381 				}else{
382 					this.el.hide();
383 					this.el.css({
384 						visibility:"hidden"
385 					});
386 				}
387 			}
388 			
389 			var lastVisible = this.visible;
390 			this.visible = visible;
391 			
392 			if(lastVisible != visible && !silent){
393 				this.fireEvent("visibilityChange",{},false);
394 			}
395 		},
396 		getVisible:function(){
397 			if(this.el.is(':hidden')){
398 				return false;
399 			}else{
400 				return true;
401 			}
402 		},
403 		getZIndex:function(){
404 			return this.zIndex;
405 		},
406 		/**
407 		 * @public
408 		 * @description 获取该component的X坐标。
409 		 * @return {Number} X坐标
410 		 */
411 		getX:function(){
412 			var left = this.el.offset().left;
413 			if(this.position == "fixed"){
414 				return left - jQuery(window).scrollLeft();
415 			}else{
416 				return left;
417 			}
418 		},
419 		/**
420 		 * @public
421 		 * @description 获取该component的Y坐标。
422 		 * @return {Number} Y坐标
423 		 */
424 		getY:function(){
425 			var top = this.el.offset().top;
426 			if(this.position == "fixed"){
427 				return top - jQuery(window).scrollTop();
428 			}else{
429 				return top;
430 			}
431 		},
432 		/**
433 		 * @public
434 		 * @description 设置该component在浏览器中的坐标及层级顺序。
435 		 * @param {Number} [x] X坐标
436 		 * @param {Number} [y] Y坐标
437 		 * @param {Integer} [zIndex] 层级顺序索引
438 		 * @param {Boolean} [triggerEvent] 是否触发move事件。默认为true.
439 		 * @param {Boolean} [useOffset] 若为ture，则会根据该component当前的position计算出top和left值，是的component在浏览器中的位置与设定的x，y值保持一致。若为false，则仅将x,y值赋值为元素的left和top属性值。默认为true。
440 		 */
441 		setPosition:function(x,y,zIndex,triggerEvent,useOffset){
442 			if(!Alipw.isSet(triggerEvent)){
443 				triggerEvent = true;
444 			}
445 			if(!Alipw.isSet(useOffset)){
446 				useOffset = true;
447 			}
448 			
449 			if(Alipw.isSet(x)){
450 				if(this.position == "fixed"){
451 					if(parseInt(Alipw.ie()) == 6){
452 						this.el.css("left", (x + jQuery(window).scrollLeft()) + "px");
453 						this._ie6_fixedX = x;
454 					}else{
455 						this.el.css("left", x + "px");
456 					}
457 				}else{
458 					if(useOffset){
459 						this.el.offset({left:x});
460 					}else{
461 						this.el.css({left:x});
462 					}
463 				}
464 			}
465 			if(Alipw.isSet(y)){
466 				if(this.position == "fixed"){
467 					if(parseInt(Alipw.ie()) == 6){
468 						this.el.css("top", (y + jQuery(window).scrollTop()) + "px");
469 						this._ie6_fixedY = y;
470 					}else{
471 						this.el.css("top", y + "px");
472 					}
473 				}else{
474 					if(useOffset){
475 						this.el.offset({top:y});
476 					}else{
477 						this.el.css("top", y + "px");
478 					}
479 				}
480 			}
481 			
482 			if(Alipw.isSet(zIndex))this.el.css("z-index",zIndex);
483 			
484 			if(triggerEvent){
485 				this.fireEvent("move",{},false);
486 			}
487 		},
488 		/**
489 		 * @public
490 		 * @description 设置该component为可用。
491 		 */
492 		enable:function(){
493 			this.enabled = true;
494 		},
495 		/**
496 		 * @public
497 		 * @description 设置该component为不用。
498 		 */
499 		disable:function(){
500 			this.enabled = false;
501 		},
502 		/**
503 		 * @public
504 		 * @description 重新计算该component的布局。此方法通常在component的布局发生变化或需要更新时自动执行，在复杂的界面控件中，此方法可能会占用较多的计算资源，若需要手动调用，请谨慎使用。
505 		 */
506 		doLayout:function(buffer){
507 			var _this = this;
508 			
509 			var layoutAction = function(){
510 				_this.fireEvent('beforeLayout',{},false);
511 				_this._doLayout();
512 				_this.fireEvent('afterLayout',{},false);
513 			};
514 			
515 			if(this.__doLayoutBuffer)clearTimeout(this.__doLayoutBuffer);
516 			if(buffer){
517 				this.__doLayoutBuffer = setTimeout(layoutAction,1);
518 			}else{
519 				layoutAction();
520 			}
521 		},
522 		//protected
523 		_doLayout:function(){
524 			
525 		},
526 		//private
527 		_runAutoRender:function(){
528 			if(this.autoRender){
529 				this.render();
530 			}
531 		},
532 		//private
533 		afterLayoutHandler_Component:function(e){
534 			if(this.parentContainer){
535 				this.parentContainer.doLayout(true);
536 			}
537 		},
538 		//private
539 		scrollHandler_Component:function(){
540 			var jWin = jQuery(window);
541 			if(this.position == "fixed"){
542 				if(parseInt(Alipw.ie()) == 6){
543 					this.el.css({
544 						left:jWin.scrollLeft() + this._ie6_fixedX,
545 						top:jWin.scrollTop() + this._ie6_fixedY
546 					});
547 				}
548 			}
549 		},
550 		//private
551 		creationCompleteFn_Component:function(){
552 			if(this.draggable){
553 				var customizedDragConfig = typeof(this.draggable) == "object"?this.draggable:{};
554 				this.dragSupporter = new Alipw.DragSupporter(jQuery.extend({
555 					target:this
556 				},customizedDragConfig));
557 				this.addEventListener('destroy',function(){
558 					if(this.dragSupporter){
559 						this.dragSupporter.destroy();
560 						this.dragSupporter = null;
561 					}
562 				},this);
563 				
564 				this.addEventListener("dragstart",this.dragStartHandler_Component,this);
565 				this.addEventListener("dragend",this.dragEndHandler_Component,this);
566 			}
567 			
568 			if(this.tooltip){
569 				var customizedToolTipConfig;
570 				if(typeof(this.tooltip) == "object"){
571 					customizedToolTipConfig = this.draggable;
572 				}else if(typeof(this.tooltip) == "string"){
573 					customizedToolTipConfig = {html:this.tooltip};
574 				}
575 				this.tooltip = new Alipw.ToolTip(jQuery.extend({
576 					target:this
577 				},customizedToolTipConfig));
578 			}
579 		},
580 		//private
581 		dragStartHandler_Component:function(){
582 			if(!this.dragSupporter.useProxy){
583 				if(this.shadow){
584 					this.shadow.disable();
585 				}
586 			}
587 		},
588 		//private
589 		dragEndHandler_Component:function(e,ep){
590 			if(this.shadow){
591 				this.shadow.enable();
592 			}
593 		},
594 		//private
595 		mouseOverHandlerForOverCls_Component:function(e){
596 			if(this.destroyed)return;
597 			this.el.addClass(this.overCls);
598 		},
599 		//private
600 		mouseOutHandlerForOverCls_Component:function(e){
601 			if(this.destroyed)return;
602 			this.el.removeClass(this.overCls);
603 		},
604 		//private
605 		mouseDownHandlerForDownCls_Component:function(e){
606 			if(this.destroyed)return;
607 			this.el.addClass(this.downCls);
608 		},
609 		//private
610 		mouseUpHandlerForDownCls_Component:function(e){
611 			if(this.destroyed)return;
612 			this.el.removeClass(this.downCls);
613 		}
614 	};
615 	
616 	return pub;
617 }();
