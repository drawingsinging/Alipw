1 /**
  2  * @class
  3  * @type singleton
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 类管理器。用于管理类的加载及加载状态的监视，通常不需要直接使用。
  6  */
  7 Alipw.ClassManager = function(){
  8 	var classes = {
  9 		'Alipw.ClassManager':'loaded',
 10 		'Alipw.Component':'loaded',
 11 		'Alipw.ComponentManager':'loaded',
 12 		'Alipw.BoxComponent':'loaded',
 13 		'Alipw.Shadow':'loaded',
 14 		'Alipw.Container':'loaded',
 15 		'Alipw.Nonvisual':'loaded',
 16 		'Alipw.Template':'loaded',
 17 		'Alipw.Module':'loaded',
 18 		'Alipw.Event':'loaded',
 19 		'Alipw.EventManager':'loaded',
 20 		'Alipw.WinProxy':'loaded'
 21 	};
 22 	
 23 	var classRequires = {
 24 		'Alipw.Container':['Alipw.BoxComponent'],
 25 		'Alipw.Panel':['Alipw.Container','Alipw.Button'],
 26 		'Alipw.Window':['Alipw.ResizeSupporter','Alipw.DragSupporter','Alipw.Panel','Alipw.ScreenMask','Alipw.WindowManager','Alipw.AnimationProxy'],
 27 		'Alipw.Taskbar':['Alipw.Window','Alipw.TaskbarItem'],
 28 		'Alipw.Button':['Alipw.BoxComponent'],
 29 		'Alipw.Msg':['Alipw.Window','Alipw.Template'],
 30 		'Alipw.ScreenMask':['Alipw.FixedWrapper'],
 31 		'Alipw.FixedWrapper':['Alipw.Container'],
 32 		'Alipw.ResizeSupporter':['Alipw.BorderContainer'],
 33 		'Alipw.BorderContainer':['Alipw.Container'],
 34 		'Alipw.ColorPicker':['Alipw.BoxComponent'],
 35 		'Alipw.Gallery':['Alipw.DataProxy'],
 36 		'Alipw.Slider':['Alipw.Component'],
 37 		'Alipw.Scrollbar':['Alipw.Slider'],
 38 		'Alipw.ValidatorGroup':['Alipw.Validator'],
 39 		'Alipw.List':['Alipw.BorderContainer','Alipw.ListItem','Alipw.DataStore'],
 40 		'Alipw.ComboBox':['Alipw.BorderContainer','Alipw.List'],
 41 		'Alipw.DatePicker':['Alipw.BorderContainer','Alipw.utils.Date'],
 42 		'Alipw.ToolTip':['Alipw.BorderContainer']
 43 	};
 44 	
 45 	var classRoot = Alipw.classPath;
 46 	var createClassScript = function(cls){
 47 		classes[cls] = "loading";
 48 		
 49 		Alipw.loadScript({
 50 			url:classRoot + cls + ".js",
 51 			success:function(){
 52 				classes[cls] = "loaded";
 53 				var defaultConfig = Alipw.Config.get(cls);
 54 				if(defaultConfig){
 55 					var classObject = Alipw.getObjectByName(cls);
 56 					if(classObject){
 57 						if(classObject.prototype){
 58 							Alipw.apply(classObject.prototype,defaultConfig);
 59 						}else{
 60 							Alipw.apply(classObject,defaultConfig);
 61 						}
 62 					}
 63 				}
 64 				if(pub.getStatus() == "complete"){
 65 					pub.complete();
 66 				}else{
 67 					updateLoadingProcess();
 68 				}
 69 			}
 70 		});
 71 	};
 72 	var updateLoadingProcess = function(){
 73 		for(var i in classes){
 74 			if(classes[i] == "pending"){
 75 				createClassScript(i);
 76 				return;
 77 			}
 78 		}
 79 	};
 80 
 81 	var pub = 
 82 	/** @lends Alipw.ClassManager */
 83 	{
 84 		/**
 85 		 * @public
 86 		 * @description 获取当前类的加载状态。
 87 		 * @return {String} 返回'loading'或'complete'
 88 		 */
 89 		getStatus:function(){
 90 			for(var i in classes){
 91 				if(classes[i] != "loaded"){
 92 					return "loading";
 93 				}
 94 			}
 95 			
 96 			return "complete";
 97 		},
 98 		loadClass:function(){
 99 			var currentStatus = pub.getStatus();
100 			pub.addClass.apply(pub,arguments);
101 			if(currentStatus == "complete"){
102 				updateLoadingProcess();
103 			}
104 		},
105 		addClass:function(){
106 			var pendingClasses = arguments;
107 			for(var i=0,len=pendingClasses.length;i<len;i++){
108 				if(classes[pendingClasses[i]]){
109 					continue;
110 				}
111 				var requires = classRequires[pendingClasses[i]];
112 				if(requires instanceof Array){
113 					pub.addClass.apply(pub,requires);
114 				}
115 				
116 				classes[pendingClasses[i]] = "pending";
117 			}
118 		},
119 		complete:function(){
120 			Alipw.classLoadedHandler();
121 		}
122 	};
123 	
124 	return pub;
125 }();
