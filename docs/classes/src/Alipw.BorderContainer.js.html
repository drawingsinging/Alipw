1 /**
  2  * @constructor Alipw.BorderContainer
  3  * @extends Alipw.BoxComponent
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 带边框的盒状控件。支持九宫格皮肤。
  6  */
  7 
  8 Alipw.BorderContainer = Alipw.extend(Alipw.Container,
  9 /** @lends Alipw.BorderContainer.prototype */
 10 {
 11 	/**
 12 	 * @property
 13 	 * @type String
 14 	 * @default 'alipw-bordercontainer'
 15 	 * @description [config option]HTML元素的class名称的基本前缀。
 16 	 */
 17 	baseCls:"alipw-bordercontainer",
 18 	constructor:function(){
 19 		Alipw.BorderContainer.superclass.constructor.apply(this,arguments);
 20 	},
 21 	initialize:function(){
 22 		Alipw.BorderContainer.superclass.initialize.apply(this,arguments);
 23 	},
 24 	commitProperties:function(){
 25 		Alipw.BorderContainer.superclass.commitProperties.apply(this,arguments);
 26 	},
 27 	createDom:function(){
 28 		Alipw.BorderContainer.superclass.createDom.apply(this,arguments);
 29 		//this.el.append(['<div class="' + this.baseCls + '-wrap"></div>'].join(""));
 30 		
 31 		this.el.append(new Alipw.Template([
 32 			'<div class="{$baseCls}-wrap">',
 33 	    		'<div class="{$baseCls}-header">',
 34 		    		'<div class="{$baseCls}-header-left">',
 35 			    		'<div class="{$baseCls}-header-right">',
 36 				    		'<div class="{$baseCls}-header-center">',
 37 				    		'</div>',
 38 			    		'</div>',
 39 		    		'</div>',
 40 	    		'</div>',
 41 	    		'<div class="{$baseCls}-body">',
 42 		    		'<div class="{$baseCls}-body-left">',
 43 			    		'<div class="{$baseCls}-body-right">',
 44 				    		'<div class="{$baseCls}-body-center">',
 45 					    		'<div class="{$baseCls}-body-content"></div>',
 46 				    		'</div>',
 47 			    		'</div>',
 48 		    		'</div>',
 49 	    		'</div>',
 50 	    		'<div class="{$baseCls}-footer">',
 51 		    		'<div class="{$baseCls}-footer-left">',
 52 			    		'<div class="{$baseCls}-footer-right">',
 53 				    		'<div class="{$baseCls}-footer-center">',
 54 				    		'</div>',
 55 			    		'</div>',
 56 		    		'</div>',
 57 	    		'</div>',
 58 	    	'</div>']
 59 		).set({baseCls:this.baseCls})
 60 		.compile());
 61 	},
 62 	renderComplete:function(){
 63 		Alipw.BorderContainer.superclass.renderComplete.apply(this,arguments);
 64 	},
 65 	getWidthByBody:function(width){
 66 		return width + (this.getWidth() - this.getBody().width());
 67 	},
 68 	getHeightByBody:function(height){
 69 		return height + (this.getHeight() - this.getBody().height());
 70 	},
 71 	//protected
 72 	_doLayout:function(){
 73 		Alipw.BorderContainer.superclass._doLayout.apply(this,arguments);
 74 		var body = this.getBody();
 75 		var wrap = this.el.find("." + this.baseCls + "-wrap");
 76 		
 77 		if(this.autoHeight){
 78 			this.height = null;
 79 			this.el.css('height','auto');
 80 			body.css('height','auto');
 81 			
 82 			if(this.minHeight && this.getHeight() < this.minHeight){
 83 				this.height = this.minHeight;
 84 			}
 85 			
 86 			if(this.maxHeight && this.getHeight() > this.maxHeight){
 87 				this.height = this.maxHeight;
 88 			}
 89 		}
 90 		
 91 		if(this.height){
 92 			body.height(10);//for ie6 bug -- when there's no content there the height is 1px but not 0
 93 			var heightDetla = wrap.outerHeight(true) - body.height();
 94 			body.height(this.height - heightDetla);
 95 		}
 96 		
 97 		if(this.width){
 98 			body.css('width','auto');
 99 			var widthDetla = wrap.outerWidth(true) - body.width();
100 			body.width(this.width - widthDetla);
101 		}
102 	},
103 	remove:function(){
104 		Alipw.BorderContainer.superclass.remove.apply(this,arguments);
105 	},
106 	//protected
107 	getBody:function(){
108 		if(!this.__getBodyCache){
109 			this.__getBodyCache = this.el.find('.' + this.baseCls + '-body-content');
110 			this.addEventListener('destroy',function(){
111 				this.__getBodyCache = null;
112 			},this);
113 		}
114 		return this.__getBodyCache;
115 	}
116 });
