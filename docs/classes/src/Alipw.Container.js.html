1 /**
  2  * @constructor Alipw.Container
  3  * @extends Alipw.BoxComponent
  4  * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5  * @description 容器，容器型可视控件的基类。容器型可视控件是指用于显示其它内容的控件。如：Panel,Window等。
  6  * @example
  7  * 
  8  */
  9 
 10 Alipw.Container = Alipw.extend(Alipw.BoxComponent,
 11 /** @lends Alipw.Container.prototype */
 12 {
 13 	/**
 14 	 * @property
 15 	 * @type String
 16 	 * @default ''
 17 	 * @description 显示在内容区域的HTML代码
 18 	 * */
 19 	html:"",
 20 	constructor:function(){
 21 		Alipw.Container.superclass.constructor.apply(this,arguments);
 22 	},
 23 	commitProperties:function(){
 24 		Alipw.Container.superclass.commitProperties.apply(this,arguments);
 25 	},
 26 	initialize:function(){
 27 		Alipw.Container.superclass.initialize.apply(this,arguments);
 28 		this.items = new Array();
 29 	},
 30 	createDom:function(){
 31 		Alipw.Container.superclass.createDom.apply(this,arguments);
 32 	},
 33 	renderComplete:function(){
 34 		Alipw.Container.superclass.renderComplete.apply(this,arguments);
 35 		if(this.html){
 36 			this.appendChild(this.html);
 37 		}
 38 	},
 39 	/**
 40 	 * @public
 41 	 * @param {HTML Element/Alipw.Component} o 需要被插入的对象
 42 	 * @param {Boolean} updateLayout 需要被插入的对象
 43 	 * @description 向容器中插入指定的HTML元素或Component实例
 44 	 * */
 45 	appendChild:function(o,updateLayout){
 46 		if(!Alipw.isSet(updateLayout)){
 47 			updateLayout = true;
 48 		}
 49 		
 50 		var body = this.getBody();
 51 		if(o instanceof Alipw.Component){
 52 			body.append(o.el);
 53 			this.items.push(o);
 54 			o.parentContainer = this;
 55 		}else{
 56 			body.append(o);
 57 		}
 58 		if(updateLayout){
 59 			this.doLayout();
 60 		}
 61 	},
 62 	/**
 63 	 * @public
 64 	 * @param {HTML Element/Alipw.Component} o 需要被移除的对象
 65 	 * @description 从容器中移除指定的HTML元素或Component实例
 66 	 * */
 67 	removeChild:function(o,updateLayout){
 68 		if(!Alipw.isSet(updateLayout)){
 69 			updateLayout = true;
 70 		}
 71 		var body = this.getBody();
 72 		if(o instanceof Alipw.Component){
 73 			o.remove();
 74 			o.parentContainer = null;
 75 			Alipw.removeItemFromArray(o,this.items);
 76 		}else{
 77 			jQuery(o).detach();
 78 		}
 79 		if(updateLayout){
 80 			this.doLayout();
 81 		}
 82 	},
 83 	/**
 84 	 * @public
 85 	 * @description 获取当前容器用于包含其它内容的HTML节点。
 86 	 * @return {jQuery Object} jQuery HTML元素对象
 87 	 * */
 88 	getBody:function(){
 89 		return this.el;
 90 	},
 91 	scrollX:function(length){
 92 		var container = this.getBody();
 93 		
 94 		container.scrollTop(container.scrollLeft() + length); 
 95 	},
 96 	scrollY:function(length){
 97 		var container = this.getBody();
 98 		
 99 		container.scrollTop(container.scrollTop() + length); 
100 	}
101 });
