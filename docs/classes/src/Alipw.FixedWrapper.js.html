1 /**
  2  * @constructor
  3  * @extends Alipw.BoxComponent
  4  */
  5 
  6 Alipw.FixedWrapper = Alipw.extend(Alipw.Container,
  7 /** @lends Alipw.FixedWrapper.prototype */
  8 {
  9 	floating:true,
 10 	baseCls:"alipw-fixedWrapper",
 11 	constructor:function(){
 12 		Alipw.FixedWrapper.superclass.constructor.apply(this,arguments);
 13 	},
 14 	initialize:function(){
 15 		Alipw.FixedWrapper.superclass.initialize.apply(this,arguments);
 16 	},
 17 	commitProperties:function(){
 18 		Alipw.FixedWrapper.superclass.commitProperties.apply(this,arguments);
 19 	},
 20 	createDom:function(){
 21 		Alipw.FixedWrapper.superclass.createDom.apply(this,arguments);
 22 		this.el.append(['<div class="' + this.baseCls + '-innerWrap"></div>'].join(""));
 23 	},
 24 	renderComplete:function(){
 25 		Alipw.FixedWrapper.superclass.renderComplete.apply(this,arguments);
 26 		
 27 		var resizeHandler = Alipw.createFuncProxy(this.resizeHandler_FixedWrapper,this);
 28 		var scrollHandler = Alipw.createFuncProxy(this.scrollHandler_FixedWrapper,this);
 29 		jQuery(window).bind("resize",resizeHandler);
 30 		jQuery(window).bind("scroll",scrollHandler);
 31 		this.addEventListener('destroy',function(e){
 32 			jQuery(window).unbind('resize',resizeHandler);
 33 			jQuery(window).unbind('scroll',scrollHandler);
 34 		},this);
 35 		
 36 		this.resizeHandler_FixedWrapper();
 37 		this.scrollHandler_FixedWrapper();
 38 	},
 39 	remove:function(){
 40 		Alipw.FixedWrapper.superclass.remove.apply(this,arguments);
 41 	},
 42 	getBody:function(){
 43 		return this.el.find("." + this.baseCls + "-innerWrap");
 44 	},
 45 	//private
 46 	resizeHandler_FixedWrapper:function(){
 47 		if(parseInt(Alipw.ie()) == 6){
 48 			this.setWidth(jQuery(window).width());
 49 			this.setHeight(jQuery(window).height());
 50 		}
 51 	},
 52 	//private
 53 	scrollHandler_FixedWrapper:function(){
 54 		if(parseInt(Alipw.ie()) == 6){
 55 			this.el.css({
 56 				left:jQuery(document).scrollLeft() + "px",
 57 				top:jQuery(document).scrollTop() + "px"
 58 			});
 59 		}
 60 	}
 61 });
