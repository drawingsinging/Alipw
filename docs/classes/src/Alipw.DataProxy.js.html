  7 Alipw.DataProxy = Alipw.extend(Alipw.Nonvisual,
  8 /** @lends Alipw.DataProxy.prototype */
  9 {
 10 	/**
 11 	 * @property
 12 	 * */
 13 	total:1,
 14 	/**
 15 	 * @property
 16 	 * */
 17 	limit:1,
 18 	/**
 19 	 * @property
 20 	 * */
 21 	indexFrom:0,
 22 	/**
 23 	 * @property
 24 	 * */
 25 	limitParamName:"limit",
 26 	/**
 27 	 * @property
 28 	 * */
 29 	pageParamName:"page",
 30 	constructor:function(){
 31 		Alipw.DataProxy.superclass.constructor.apply(this,arguments);
 32 	},
 33 	commitProperties:function(){
 34 		Alipw.DataProxy.superclass.commitProperties.apply(this,arguments);
 35 	},
 36 	initialize:function(){
 37 		Alipw.DataProxy.superclass.initialize.apply(this,arguments);
 38 	},
 39 	/**
 40 	 * @public
 41 	 */
 42 	load:function(index){
 43 		if(this.limit * index >= this.total){
 44 			//throw
 45 			return;
 46 		}
 47 		var dpx = this;
 48 		var sendingData = new Object();
 49 		sendingData[this.limitParamName] = this.limit;
 50 		sendingData[this.pageParamName] = index + this.indexFrom;
 51 		jQuery.ajax({
 52 			url:this.url,
 53 			data:sendingData,
 54 			success:function(data){
 55 				dpx.fireEvent("complete",{
 56 					content:data,
 57 					index:index
 58 				});
 59 			}
 60 		});
 61 	}
 62 });
